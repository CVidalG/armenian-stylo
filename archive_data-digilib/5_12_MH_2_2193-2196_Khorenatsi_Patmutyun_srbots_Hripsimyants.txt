ՄՈՎՍԷՍ ԽՈՐԵՆԱՑԻՊԱՏՄՈՒԹԻՒՆ ՍՐԲՈՑ ՀՌԻՓՍԻՄԵԱՆՑ        1 Յաղագս զանազան տեղեաց եւ գնացից եւ վարուց սրբոց հռիփսիմեանցն, որ ոչ երեւին գրեալ ի յոլովից ճոխաբան պատմութեանց, ի բազմաց մատենագրաց Յունաց արարեալ քննութիւն եւ իմացեալ զմիտս ճառից որ հեռի էր ի ստութենէ եւ լի ճշմարթեամբ, կարգեցի քեզ ի գիրս յայսմ. որում ոչ զոք կարծեմ ընդդիմանալ ի միտս ունողացն. բայց միայն եթէ ոք ստութեան հաճեալ, զոճ պատմութեանս քակել ջանայցէ։ 2 Զատենաբանութիւնս իմաստնոց յելլադացւոց յԱտտիկէ իմ ընթերցեալ, բազում անգամ գտի զոճ գրոցս, որ ի ճահ գայր պատմութեանս մերոյ հիւսման։ 3 Յոյժ պիտանի եւ յաւգտութիւն մեզ լսի` որ անուն գրոցն կոչի Դաւթի հռովմայեցւոյ գիրս եկլեսիաստէս. զոր կամեցեալ իմ թարգմանել, եւ ոչ ժամանեցի. մնաց այլում տեղւոյ եւ կամ իմաստնոյ։ 4 Զի ես մահկանացու գոլով, ծեր եւ հիւանդոտ, երկուցեալ ոչ հրեշտակ գոլով, կամ որդի բարձրելոյ, որպէս աստուածայինն ասէ բան, սակաւ մի երկրորդեցի ի նոյն գրոցս, զոր քաջ քարտուղարն թողեալ Ագաթանգեղոս, որ եւ ոչ կարէր զամենայն զբանս եւ զգործս ի միտսն ծովացուցանել, մնալով եւ այլոց եւս։    5 Եւ արդ ի նոյն գիրս պատմի, ի նոյն թագաւորական տոհմէ գոլով սրբոյն, լինել Աւգոստոս կայսեր` ցեղապետ թագաւորաց։ 6 Լինի եւ Գայիանէ ի նոյն ցեղէ, առաքինի եւ զարմանալի հաւր եւ մաւր դուստր։ 7 Նախանձ ունելով առաքելական դասուցն, գրէ որպէս տեսութեամբ նոքա էին մաւտագոյնք առ նոսա. որոց ոչ հին համբաւով ի մոռացումն ինչ էին եկեալ գործք նոցա, ընդելեալ կամ յուլացեալ, որպէս առ մերովք ժամանակաւքս աստուածպաշտութիւնն նուաղեալ, սրբութիւն լռեալ, եւ տնկեալ մեղք։ 8 Այլ արքն որպէս թէ առաքեալք, լաւութեան նախանձ ունէին, եւ կանայքն զգաստանային նախանձ ունելով Աստուածածնին, Քրիստոսի լինելով մարք, որպէս ասաց ոմն ի հարցն։ 9 Յայսցանէ գոլով եւ ծնողքն սրբոյն Հռիփսիմեայ, սան յաւազանէն լեալ սրբոյն Գայիանեայ, որ էր մայր կուսից վանականաց ի Հռովմ ի վանս սրբոյ առաքելոյն Պաւղոսի. զոր իւր տեղի որոշեալ երանելոյն, պատշաճ եւ պարտ գոլով, հրաման յառաքելոյն առեալ աղաւթել. որ եւ այլք յետ նորա աշակերտեցան, լինելով անձինս երեք հարիւր, պահելով զհրամանս առաքելոյն։ 10 Եւ այսմ տեղւոջ տեսուչ գոլով Գայիանեայ, սնուցանէ առ ինքեան զՀռիփսիմէ ի մէջ զգաստացելոց կուսանացն. որ եւ պտուղ աղաւթից նորա յարբունս հասեալ, տեսեալ ի նմա զհոգւոյն մեծանձնութիւնն, անուանէ զնա աղախին գոլ սրբոյ Աստուածածնին, եւ տայ զնա ուխտիւք եւ աղաւթիւք ծառայել Աստուածածնին։ 11 Արկանէ եւ ի նմին պարանոցն խաչ զսրբոյն Յակովբայ զմածեալն տէրունական արեամբն, ի կենաց փայտէն հատեալ վասն խնդրոյն Պատրոնիկեայ տիկնոջ, որ եւ ինքեան գոլ պահապան ի ձեռն նմին եղբաւր տեառն, առաւել փափաքմամբ եւ երկիւղիւ առեալ արկանէր ի պարանոց սրբոյ մեծին Հռիփսիմեայ, որպէս եւ յայնժամ յականջսն Ռեբեկայ զգինդսն, հարսնանալ հաւրն հաւատոց Աբրահամու, թէպէտ եւ ոչ նման, այլ երկոքին ի խորհուրդ եկեղեցւոյ. եւ այս այսչափ։    12 Այլ ի հասանել յուզման ժամանակին, վասն ազնուական ազգին եւ սքանչելատեսիլ գեղոյն, որ եւ յաշխարհի փայլէր հոգւով, խորհէին առնուլ ի պատիւ տանն հռոմայեցւոց ի թագուհութիւն տիեզերապետութեան։ 13 Եւ Գայիանեայ իմացեալ հոգւոյ ակամբ, առեալ զնա փախչի յԵգիպտոս յԱղեքսանդրիա. եւ անտի ելանէ յԵրուսաղէմ երկրպագեալ սրբոյ տեղեացն որ անդ։ 14 Եւ երթեալք ի յայրն, աղաւթս եւ պաղատանս մատուցանէին առաջի սրբուհւոյն, աւգնել կուսին վտանգելոյ եւ փախուցելոյ, որ եղեւ նորա աղախին ի մանկութենէ։ 15 Եւ գնացեալ անտի ի Գեթսեմանի ի գերեզման Աստուածածնին, եւ անդ աղաւթելով ասէր. Մայր լուսոյ կոյս անարատ, որ կուսութեանց ես վերակացու, եւ սրբութեան աւգնական, ընկալ զաղաչանաց աղաւթս յանարժան աղախնոյս. եւ զորդի հոգւոյ իմոյ զոր սնուցի քեզ տուեալ աղախին, զգաստացեալ արդարացո արժանաւորապէս, եւ մի' տար փորձել աւտարին, եւ մի' տացես աղտեղանալ անձամբ ÿի ձեռն պիղծ թագաւորացն, զի մի' ուխտս մեր որ առ քեզ է, խափանեսցի։ 16 Այլ յիշեա զպանդխտութիւն իմ եւ զծուփս ի կենցաղս մեր որ վասն քո, յիշեա եւ զչարչարանս որդւոյ քո եւ Աստուծոյ մերոյ, եւ ողորմեա մեզ. վասն որոյ զքեզ աւրհնեմք։ 17 Եւ երկրպագելով Աստուածածնին եւ բազում արտասուս հեղեալ, մնալով անդ ամսաւրեայ ժամանակաւ։ 18 Ապա տեսիլ երեւեալ սրբոցն, եւ հրաման առեալ գնալ ի Հայք յերկիրն Արարատեան ի բաժին Թադէոսի առաքելոյն։ 19 Եւ զպատկերն փայտեղէն որ եդեալ է յերեսս իմ յաւուր փոխման իմոյ, զնա խնդրեսջիք, կատարեալ զընթացս ձեր արեամբ։    20 Եւ դարձեալ անտի խնդութեամբ եկին յԵդեսիա քաղաք, եւ երկրպագեալք փրկչական պատկերին` լցան ուրախութեամբ, համարեցան տեսեալ զԲանն մարմին եղեալ։ 21 Դարձեալ սքանչելի տեսիլ երեւեալ յորդորէին ի ճգունս։ 22 Եւ այլ կանայք ի նոցանէ մեկնեալք մնացին ի նմին քաղաքի ի մեծ եկեղեցւոջն Աբգարու, ի հիման սրբոյն Թադէոսի կրաւնաւորեալք։ 23 Սապէս եւ անտի ի Դարանաղեաց գաւառին կացեալ ի խորշ լերինն Սեպուհ կոչեցեալ, որ անուանեցաւ Մանեայ այրք։    24 Աւելի քան զեաւթանասուն հոգի զսրբոցն ասէ պատմագիրս ելեալ ընդ նոսա ի հռովմայեցւոց քաղաքէն, որք բնակեցան ի տեղիս տեղիս։ 25 Իսկ զտեղին կալեալ այլոց քերցն, եւ քոյր Գայիանեայ Եւզոյի անուն, միաբանեալ քերբքն կատարեցան բարի վկայութեամբ ի Դիոկղետիանոսէ ամբարշտէ, զթոյնս չարութեանն ի նոսա թափելով, որք եղեն վկայք Քրիստոսի, յաղագս ոչ ըմբռնելոյ ի վարկս մահու զմեծ առաքելուհին։ 26 Եւ բազում տառապանաւք գնացեալ եկին ի Բզնունեաց գաւառ, ի քաղաքն Խլաթ, երեկաւթս արարեալ։ 27 Եւ անտի փոխին, եւ գնացեալ ի գաւառն Մոկաց, Սուրբ խաչ կոչեցեալ տեղի. որ եւ մեծ սքանչելիս կատարեաց Աստուած ի ձեռն նոցա պատուական սուրբ խաչիւն Եսայեայ եւ Դանիէլի, որում ես եւ դու հասու, եւ մեզ երկրորդել զի է։ 28 Կճաւն կոչեցեալ տեղի` դիւաց լցեալ պաշտմամբ, հասեալ անդ առաքելանման քահանայիցն եւ սրբոց կուսանացն, զընդդիմամարտն արգելուին զձիոցն. յորում զայրագնեալ դիւացն` դիմեալ ի վերայ խեղէին, ծռէին, պնչատէին զպաշտաւնեայս իւրեանց. որ եւ անդ առաքելանշան քահանայքն հալածեալ զնոսա, եւ քանդեալ զպաշտամունս նոցա, բժշկէին զխեղեալսն. վասն որոյ տեղին կոչեցաւ Կճաւ. յորում թողու զխաչն իւր սուրբն Գայիանէ ախոյեան դիւաց` որ էր աւազակին. որ եւ զտեղին յետոյ հաստատէ վանս սուրբն Գրիգոր։    29 Եւ վասն յայտնի անուանն նոքա գաղտ գնացեալ ընդ մէջ Կորդուաց ընդ լեառն Սողոփ, ի Բերկրացւոց գաւառին եւ ախտ աչացաւութեան զնոսա տագնապեալ, եւ կացեալ յաղաւթս, երկու աղբերք բղխեցան, որ է ժանգ ջրոյն կարմիր եւ սպիտակ. եւ մինչեւ ցայսաւր աւգնէ աչացաւաց յանուն սրբուհեացն։ 30 Բայց այսր լերին ասի յԱսորւոց, թէ ի նուազել ջուրցն հեղեղաց, եհաս տապանն ի գլուխ լերինն, որ է Սարարադայ. եւ ընդ մէջ անցեալ սղոցաձուկնն արգելու զնաւն. եւ անուն քաղաքագեղջն կոչեցաւ Թըմնիս, այսինքն թէ ութ ոգիք ելին ի տապանէն։    31 Իսկ սրբոցն յապահովացեալք, խրճիթս շինեալ բնակեցան։ 32 Եւ անտի գնացեալ եւ աւթեվանս արարեալ ի վտակս գետոյն մաւտ ի կամուրջն ի կողմն Տմորեաց, ի ծառաւէտ ընկուզուտ տեղիս։ 33 Եւ անտի փոխեալ ելանեն ի Հայաստան գաւառ, ի խորին խորշ լերին Մշկունեաց։ 34 Ի սմա ասի այծեմունս մշկապորտս լինել, ուր եւ բազում ուրուկս հալածականս գտեալ, բժշկեն զնոսա սուրբ խաչիւն ընտրեալ քահանայքն, ուր եւ ի տեառնէ աղբիւր շնորհեալ նոցա. եւ յայսմ տեղւոջ կային ծեր կանայքն մինչեւ ցաւր կատարման իւրեանց։ 35 Եւ չուեալ անտի եւ կացեալ յեզր Բժունեաց, կային անդ բազում աւուրս. եւ անտի փոխեալ բնակեցան յայնկոյս գետոյն մաւտ ի յակն յորդաբուղխ, յորդորեալք ի շնորհաց Սուրբ Հոգւոյն, բազում աւուրք անցուցանելով, զուարճացեալք ի տեղւոյն բարեհամբաւութիւնն։ 36 Եւ ապա աստ այլք ի նոցանէ կացեալք. որ անուանի տեղին Սուրբ տիկին։    37 Եւ անտի յառաջ անցեալ ուստի հոսի գետն բովանդակ. որ փոքրագոյն չար ջուրքն հոսեալ ի լերանց Կանգուարայ, որ կոչի Ագռաւաքար. եւ գարնանային ուղխքն գետացեալ, գան յաւժընդակութիւն գետոյն մեծի, որ կոչի Տիգրիս, հասին ի տեղի մի, եւ գտին սակաւ ինչ քրիստոնեայս` ձեւանշան ունելով զպատկեր Տիրամաւրն, որում երկրպագեցին ուրախացեալք, եւ տային գոհութիւն սրբուհւոյն։ 38 Եւ սակաւ աւուրս կացեալ, ոչ կամեցեալ հաւատացեալքն, երկուցեալք ի կասկածանացն, եւ նոքա գնացին ի գլուխ լերինն Պաղատոյ, զոր ասէին սաստիկ յոյժ ի նմա լեալ դիւացն, տուն Արամազդայ եւ Աստղկայ մեծարէին։ 39 Եւ յաճախ պաշտամամբք տաւն կարդային, որ է Պաղատ։ 40 Եւ անտի գնացեալք ի Խորաձոր զառ ի վայր լեռնէն յարեգականն կոյս, սակաւ մի առ ի կողմն հարաւոյ, ի կատար լերինն, որ ի վայր իջանէ, տուն կրակի, անյագ հրոյ, անդադար այրման աստուածոցն. եւ քարինս հոսեալ յակն յորդաբուղխ աղբերն, որ բղխէ յոտն լերինն, որ է յառաջս խորավիմին, որ կոչի տեղին Բութ։ 41 Քանզի վառէին զկրակն քոյրն, եւ զաղբիւրն եղբայրն։    42 Եւ յայրս քարին ողջացեալ կային վիշապք երկու, դիւացեալք եւ սեւացեալք, որոց աղջիկ կոյս եւ պատանիս անմեղս զենուին։ 43 Այնոցիկ արեամբն զուարճացեալ դեւքն եւ բագնաւքն, հրովն եւ աղբերբն, ահագին իմն տեսիլ եւ փայլումն եւ թնդմունս եւ կայթմունս գործէին։ 44 Եւ ի մէջ Խորաձորոցն աւձք եւ կարիճք լի թիւնաւոր մահաբերին լցեալք։ 45 Անդ հասեալ սուրբ կանանցն, եւ աղաւթիւք եւ արտասուաւք ի յերկինս հայեցեալ, առեալ զսուրբ նշանն եւ ի վերայ կացեալ աղբերն, եւ փախստական եղեալ դիւացն ի բագինն անկանէին, վայ զվայիւ աղաղակեալ։ 46 Եւ ի լեռնէ ի լեառն ոչ այլ ինչ լսիւր, բայց վայ զվայիւ ճիչեալ։ 47 Եւ նոքա ընդ մէջ ձորոյն անցեալ, աւձք մարմնաւորք քարացեալք կան մինչեւ յայսաւր։ 48 Եւ ի բագնացն փախուցեալք դեւքն անկանէին յերկիրն Մարաց։ 49 Եւ զպատկերն փշրեալ զաւրութեամբ սուրբ խաչին. զոր տեսեալ գեղջն, հաւատացին ի Քրիստոս ի ձեռն սուրբ երանելեացն։    50 Եւ անտի ելեալ եւ ընդ հիւսիսով իջեալ, եկին ի գեաւղն որ կոչի Կորիթ։ 51 Զկուռսն փշրեցին, եւ զաւձսն թիւնաւորս սպանին, նշանաւ սուրբ խաչիւն, եւ զաղբիւր դիւացն ցամաքեցուցին, զոր եւ յետոյ զտեղին սուրբն Գրիգոր շինեաց եկեղեցիս երկուս։ 52 Եւ ի Կորիթ փայտեղէն սեղան կազմէ, եւ նշան թողու անդ զգաւտին եւ զգաւազանն իւր։ 53 Իսկ սրբոցն ի Տոսբ գաւառ ելեալ, զսուրբ նշանն թողուն հանդէպ Վանի, ի հարաւակուսէ լերինն, փոքրագոյն խրճիթս շինեալ վանաց. եւ եդին անդ վասն կասկածի հալածանաց ի վերայ կացելոց։ 54 Եւ անդ ի լեառն Վարագ ամրացեալք. զի լուան խնդիր լինել ինքեանց, ցան եւ ցիր ի խորշ լերինն բնակեցան, արարեալ իւրեանց գետնափոր դադարս։ 55 Այսպէս երկուք ոմանք ի քահանայիցն սրբոց բնակեցան ի նմին տեղւոջ մինչեւ ցաւր կոչման եւ կատարման իւրեանց առ Քրիստոս։ 56 Եւ տեղիք նոցա կոչին Ուրբաթայրք. քանզի ի յուրբաթէ յուրբաթ մատուցանէին զսրբութեամբ պատարագն, եւ բաշխէին արժանաւորապէս սուրբ կանանցն զմարմին եւ զարիւնն տեառն։    57 Արդ ով պատմեսցէ զզաւրութիւնս տեառն, լսելի լինել սքանչելեաց նորա։ 58 Որ ի ձեռն ողորմութեան իւրոյ ծագեաց մեզ լոյս գերազանց յարեւմտից` արեւելից Հայաստան երկրի, մանաւանդ ÿի ձեռն կանանց, յորոց մտին մեղք յաշխարհ։ 59 Որ ամենայն ինչ կարելի է նմա, եւ տկարանայ առ ի նմանէ եւ ոչ ինչ։ 60 Արդ վասն կատարման եւ չարչարանաց եւ նահատակութեան երանելեացն, յԱյրարատեան գաւառ ի Վաղարշապատ քաղաքի ուսցիս ի յԱգաթանգեղայ պատմութեանցն։ 61 Յորոց ձեռն ծագեաց մեզ անստուեր արեգակն ի բարձանց, լուսաւորելով զմեզ լուսով աստուածգիտութեամբ. պայծառացուցանելով զսիրտս հաւատացեալ ժողովրդոց տաւնասիրաց եւ աստուածասիրաց. վասն զի Քրիստոսի վայելէ փառք եւ զաւրութիւն եւ իշխանութիւն յաւիտեանս յաւիտենից.
