հոլանէ, զի փոքրն անճառութեամբ ուսուցանելով պատմել, կարճեսցէ ի վեհագոյն պատճառէն, զտարապատկան խնդիրս յուզել, բաւական համարելով զաստուածաշունչ գրոց զյայտարարութեան կէտն։    4 Արդ, զի զամենայն զստեղծելոցն ճշդիւ եւ համառաւտաբար կատարէ զվերծանութիւնն, ազդարարական շնորհիւ Սուրբ Հոգոյն, երեւելի է. բայց արդեամբք զշինութեանն յարմարութիւն չհամարի իրաւացի ուսուցանել. ընդէր արդեաւք, զի կարծուական եւ շարժուն վարկանի զառաջադրութիւնն։ 5 Քանզի վարժեալ էր բովանդակ եգիպտական քննութեանցն խրատուք, եւ գիտէր զգուպարս ստեղծմանցն, որ ի նմա եւ եղծից։    6 Վասն այնր ոչ ճոխաբանէ յինքենէ ինչ, կամ ի շահեալն պիտակաբարն հրահանգաց։ 7 Մանաւանդ թէ չթուացուցանէ իսկ պարգեւաւղ վերակացութեանն զգետնաքարշիցն ի կիր առնուլ վսեմախոհութիւն։ 8 Այլ անսայթաք եւ ճշմարիտ սահմանաւ աստուածային հաճոյիցն, սքանչելի նորակերտութեան ցուցանէ հաստատութիւն, եւ աններհական զնոյն մարգարէից եւ առաքելոցն, վկայեալ մատեանքն մեծարին ի տիեզերական եկեղեցիս, ըստ արժանի վայելչականն ընդունելութեանն։ 9 Բայց ունի եւ ինչ յունական հետազաւտութիւնն, եւ առ սուրբ կտակարանսն փոքր մի յանկութիւն։ 10 Քանզի ընկալան որպէս յաւետախաղաց աղբերէ, յերջանիկն Մովսէսէ յաւգուտ ի տիրական պարապմանցն իւրեանց, եւ է, զոր քատակելով իմաստասիրութեանն, հաւանականս խաւսեցան։ 11 Զի զմարդ ի չորից նիւթոց բնաւորեցին՝ ի հուոյ, եւ ի ջրոյ, եւ յաւդոյ, եւ ի հրոյ, յաղագս ջերմութեան եւ ցրտութեան, եւ խոնաւութիւն, եւ ցամաքութիւն ի նմա համեմատելով. եւ զանձն եռամասն, եւ գոյ հնար դատուութեամբն յաւելուլ եւ պակասեցուցանել ըստ մասանցն, որ ի շնչականսն եւ ի զգայուն կենդանոջն եւ ի բանականումն եւ յանհատին հարցափորձի բնութեան անուամբն յեղանակելով։ 12 Զի այսքանեացս կատարելութիւն մարդոյն, ընդէր ասի մի բնութիւն. զի առ հասարակ սեռին անստգտանելի է սահմանդ։ 13 Իսկ ի վերայ անհատին առաւել պարսաւելի, եւ ե՜ւս յոլովիւքն մաքառիցիմք։ 14 Քանզի մարմին իսկ չէ հոգի, եւ ոչ անձն՝ մարմին եւ կիրք, այլ եւ այլ երկաքանչիւրոցն, եւ անզանգ եւ անշփոթ է կցորդութիւնն, եւ յայտնի է այն ի բաժանմանն։ 15 Այլ զի անճեղքելի եւ անանջատ պատկանեալ յարդարեցաւ տեսակն, թերեւս անբաղայաբար կոչեցաւ։ 16 Մի զի յիմանալի ի բա՜ց արձակութենէն, զարմազան ունի անբաժանութիւն, եւ ի զատուցմանն ոչ եւս կարէ անուանիլ նոյն։    17 Այսպիսի ուրեմն գաղափարաւ եւ արարչական Բանին իմանի տնաւրէնութիւնն։ 18 Թէպէտ եւ ոչ զաւրէն հասու լինելով միտք մեր, զի համանգամայնիւքն գերազանցեալ եւ անյա՜գ է առեցումն ի մէնջ անչափ իսկութեանն. չգո՜յ հուն յերկուանալ ումեք, այլ զի ա՜նդր ե՜ւս քան զհրաշս է խոնարհելն, զծառային ընդունելով պատկեր՝ ոչ յեղլով կամ սպառելով ի բաց բարձութեամբ միոյն զմիւսն, այլ անմեկնելի հաղորդութեամբն։ 19 Վասն որոյ ի դէպ եւ յոյժ ճահուագոյն թուի խոստովանուաց մի բնութիւն ասելով. զի եւ յաջմէ նստեալն համեմատութեամբ մի Տէր քարոզի։ 20 Ապա եթէ հակառակս ինչ վարկանիցին ոմանք, զպատասխանեալդ անկարելի համարելով, որպէս թէ երկուս բնութիւնս պատշաճ իցէ ասել, գիտասցեն զի զնոյն զանհնարութիւն եւ ի մարդումն գոյ հնար նկատելով ճանաչել ոչ միայն ի փիլիսոփայական ստորոգութիւնսն, այլ եւ սուրբ գիրք աստուածաշունչք։ 21 Թէպէտ եւ բնութեանց վարդապետութեամբ ոչ վարին, որպէս նոյն ինքն տեղեկացուցանեն պատմութիւնքն, սակայն պատուականն յաւէտ յարգանաւք, եւ արհամարհքն բազում միջոց ունին տարբերութեամբ։ 22 Զի մարմին յանարգ կաւոյ ստեղծանիլ զեկուցին՝ եւ շունչ եւ հոգի անեղին փչումն։ 23 Ծանուցանէ եւ նուիրականն առաքեալ ստուգիւ գիտելով զիւրաքանչիւրոցն զանազանութիւնս, եւ աղաւթէ՝ Անբիծ պահել զհոգի եւ զմարմին եւ զշունչ մինչեւ ի գալուստն Տեառն։ 24 Լուսաւորագոյնս եւ Փրկչին վարդապետութիւն յայտնէ՝ զհոգին առաւել ասելով քան զմարմինն, եւ այլք բազմապատիկք յաղագս նորին։    25 Արդ զո ի նոցանէ կանգնիցէ ընկեր, զերկուս կամաւղսն ցուցանել բնութիւնս Ապուղինարի լինելով ջատագով եւ նորին հետեւեցելոց, որ են աղճատանք, եւ շամբուշ յիմարութեան ձեռնարկութիւն։ 26 Զի թէ չէր ըստ պատկերի գումարութեանն մի, բոլորեցունց ծանաւթանայ այլ եւ այլ գոլ։ 27 Բայց թերեւս ասիցեն, եթէ մարմնոյ կազմութիւն արարածական է՝ եւ թէպէտ յոգունց գոյ հնար, մի ասել։ 28 Իսկ անբաղդատելի արարաւղին ո՜չ եւս նոյնպէս։ 29 Թէ այդպէս իցէ, ուրանալ պարտին եւ զմարդանալոյն վկայութիւնս։ 30 Բանն մարմին եղեւ, ասէ, եւ զծառայի առնու զնմանութիւն, զի զայն ուսուցանէ առնուլ՝ զոր ոչն ունէր յոր չբռնադատեցա՜ւն յումեքէ, եւ ոչ ամաւթ վարկաւ. յոր չէաք յանգ եղբայրութեան, եղբարս անուանեաց։ 31 Ապա թէ այնչափ անկարելի համարեսցին զմիաւորութիւնն, պարսաւելով զքաղցրութեամբն խոնարհաւղ յանշփոթութիւն եւ զանգումն փաստաբանելով մի եւս վարեսցին ձայնիւ միաւորութեանն՝ մանաւանդ դիմացն։ 32 Որ ի կերպարանս Աստուծոյ էր, ասէ, զկերպարանս ծառայի էառ։ 33 Տեսանես, զի կերպ եւ կերպ ասէ, զոր արդեաւք կերպարան զանգեալ եղծանիցեն, ըստ նոցա դաւանութեանն, զի թէ զբովանդակին հպաւորութիւն խառնակումն պատճառին՝ զնոյն եւ դիմացն պարտին իմանալ. ծաղո՜ւ արժանի են բանքն. զպատուականն, ոչ զանգիտես, ըստ նոցա աղմկեալ եւ խառնափնդորեալ վարկի՝ յանձն առնուլ, եւ զայլսն եւս սարսափեալ քստմնին. զի ըստ առասպելական բարբաջմանցն, զգլուխն մի գոլ ստեղծցեն եւ զտտունս երկաքանչիւր։    34 Աւա՜ղ թշնամանացն, որ լի է դառնագոյն ամպարշտութեամբ. մինչդեռ զմեզ բամբասէին, տխմարս եւ հայհոյիչս գոլ յաղագս զմարդկայինն նաւթճել անզատանելի, ինքեանք ոչ զտնտեսութեանն լոկ պատառեալ անջատեցին զսքանչելի փրկութեանն անքակութիւն, այլ զնորին իսկ էակից Հաւր Բանի բնութիւն։ 35 Քանզի զԱստուածային զդէմն, այլայլելի բացայայտեցին, միաւորութեամբն զանգեալ եւ շփոթեալ, որպէս իւրեանց լեզուքն վկայեն, եւ զայլն ամբողջ պահել, զի առանց կերպարանի հնարեցին ցուցանել զանձնաւորութիւնն Որդւոյ։ 36 Նոյնպէս եւ զմարդոյն՝ ի խելայեղ բարբաջանս ընկղմին, եւ զյաւիտենական կորուստն ժառանգեն, զայսպիսի սոսկալի պղտորութիւնս զգայռելով։ 37 Ոչ գիտեմք ուստի եկին յայս, զի հաւատ, որ ընդ տիեզերս տարածեալ բարեբանի, յետ զմարմին եւ զհոգի եւ զմիտս ստանալոյ Բանին, մի Տէր զՅիսուս Քրիստոս քարոզեն. ի կենդանարար բերանոյն բղխեալ պատգամս ոչ դոյզն ինչ մարդոյ, այլ որ ի ծոցոյ անսկզբնականի Հաւրն է, նմա վկայեն պատմել, եւ զսխրագործականսն եւ զաղքատութիւնն նմին մարմնովն ծագեցելում, անքակելի բարբառին։ 38 Եւ ճշմարտիւ գոյ զնոյնս յարաբարդեցելումն տեսանել. զի եթէ մոլեգնեալ ըստ ոմանց, քարկոծել ոք փութասցի, մարդ է, եւ թէ նշանս խնդրել բարեպաշտաբար յորդորիցի, Աստուած իսկապէս։ 39 Քանզի մի եւ ոչ երկուս անձնաւորութիւնս տեսին ի Տեառնէ ապաքինեալքն, եւ զերանութիւն ընկալեալք։    40 Զինչ եւ աստանաւր կամիցիս, ո՜վ անջրպետողդ. ուրեմն աստուածային դէմն ի մարդկային յանհաւասար բնութեանն միաւորեալ անարգութիւն կրեաց, միշտ մնալով յանպատուին. եւ թարց պատկերի մնացեալն զգերազանցութեանն ընկալաւ ճոխութիւն։ 41 Քստմնի՜մ յաւեժ սարսափմամբ յամպարշտութեանս յամելով թշնամանս. քանզի պարտ է կամ ըստ արժանի միաւորութեան ի բաց թուուլ զանոպայ բաժանմունս, կամ ամենեւին իսկ ուրանալ։ 42 Ընդէր իսկ կրճատել սիրէք կիսով հաղորդութեամբդ, որպէս թէ երկու կատարեալ, ոչ կարէ մի կատարեալ լինել. զայդ հարկաւորութիւն իմաստուն եւ առաւել բարգաւաճեալն հովիւ ուսոյց Բասիլիոս՝ յամաւթ առնելով զժպիրհ ախոյեանն, եւ կատարեալ անշփոթ տնաւրէնութեանն Աստուծոյն եւ մարդոյ մի բնութիւն ասել՝ զնոյն զմի զտնաւրինական եղելութիւնն։ 43 Եւ Յովհան Կոստանդնուպաւլսի այցելուն ի Փիլիպեցւոց թղթին ընդդէմ հերձուածուացն դաւանեաց. Մի շփոթեսցուք, ասէ, եւ մի անջրպետ ի մէջ արկցուք, վասն զի միութիւն խոստովանիմք, եւ ոչ քայքայութիւն ինչ եւ լուծումն այսր բնութեան յայն, կամ այնր՝ յայս, այլ միացեալ ասեմ եւ մի լեալ։ 44 Նոյնպէս եւ մեծն Յուլիոս՝ ի թղթին, որ առ երջանիկն Դիոնիսիոս Աղեքսանդրացի։ 45 Եւ Կիւրղոս յԱղեքսանդրացւոց քաղաքին պայծառացեալ՝ յերկրորդում ճառին, որ առ Սուկենոս Դիոսկուրացի, եւ յորս առ Նեստորն, եւ որ ի մեկնութեան հաւատոյ երեք հարիւրոցն, եւ Երեքթեայ երանելոյ Անտիոք Պիսիդեայ եպիսկոպոսի, ի յաստուածայայտնութիւնսն, ի Կոստանտինոյպաւլիս. եւ Գրիգորիոս Նեոյկեսարացի, յորում ըստ մասին հաւատոյն. եւ Եփրեմ ի ճառին որ ի Սուրբ Երրորդութիւնն. եւ Գրիգորիոս Նազազացի, աստուածաբանութեամբն մեծարոյ, յոլովագոյնս ասելով ժամանակի զարմանս եւ այլ անթիւ հովիւք ուղղափառութեան համաբարբառք սոցին մի Տէր, եւ մի զնորին բնութիւն գիտելով՝ անբաժ միաւորութեամբն աւետարանեալ յաշխարհի, հաւասարելով առաքելոց եւ մարգարէից պատգամաց, որք վայելչական լծակցութեամբ աւարտեն զխորհուրդ միաւորութեանն։    46 Զի աստուածափառ նախամարգարէն, զ՚ի փայտէն կախեալ կեանս կոչէ, եւ որդին Որոտման Յովհաննէս՝ ճշմարիտ Աստուած եւ կեանս յաւիտենականս։ 47 Սերովբէից տեսաւղն Եսայիաս, զ՚ի կուսէն ծնիցեալն ընդ մեզ Աստուած լինել ճոխաբանէ։ 48 Եւ Տարսոնացի փողն բարեձեւագոյնս գոչելով հնչեցուցանէ զբարբառն՝ Մի կարծէք, ասէ, ի ժամանակեան կուսէն առնուլ զսկիզբնն. զի Աստուած առաքեաց զՈրդին իւր, որ եղեւ ի կնոջէ, եւ եմուտ ընդ աւրինաւք զաւրինաւորսն ապրեցուցանելով։ 49 Եւ դարձեալ նոյն տեսանաւղ, հանդերձ միւսովն, թէ՝ Ոչ պատգամաւոր ոք եւ ոչ հրեշտակ ոք, այլ ինքնին Տէր փրկեաց զմեզ. զոր ինքնին իսկ կենարարն ուսոյց, ի խաւարէն բուժեցելում կուրին ասելով. Դու հաւատաս յՈրդի Աստուծոյ, եւ նորա զարմացեալ, զՈվն առաջի դնէր, եւ նա ասէ. Եւ տեսէր զնա, եւ որ խաւսիդ ընդ քեզ, նա է։ 50 Եւ Երեմիա ասէ. Բայ, սա է Աստուած մեր, ընդ որում ոչ համարեսցի այլ, որ ետ զիմաստութիւն Յակովբայ. յետ այսորիկ ի յերկրի երեւեցաւ եւ ընդ մարդկան շրջեցաւ։ 51 Երեւեալն յաշխարհի աղաղակէր, զնոյն հաստատելով, թէ Որ առաքեացն զիս Հայր ընդ իս է, եւ որ տեսանէ զիս, զՀայրն իմ տեսանէ, զի ես եւ Հայր իմ մի եմք։ 52 Եւ թէպէտ եւ զբազմապատիկսն եւս քննեսցուք կամ քննեսցէ ոք յաղագս այսորիկ՝ յայտարարութիւնս ի Սուրբ Գիրս՝ ոչ ուրեք գտանի երկուց բնութեանց բաժանելոց հանկանակաւ բաշխել տալ, այլում մարմնով յայտնեցելում, նոյն հովիւ եւ ոչխար, քահանայ եւ պատարագ, ճանապարհ եւ դուռն, լոյս եւ ճշմարտութիւն։ 53 Իսկ եթէ ըստ արտաքին փիլիսոփայիցն պանձացեալ, մարտնչել պէս պէս բնութեամբդ՝ թէպէտ աւտար է նոցայն բազում իւիք յաստուածապարգեւ պատուիրանէն՝ բե՜ր, աղէ՝ անցուսցուք ի հանդիսի, քուայայտելով զխղխայթեալսդ։ 54 Քանզի նշանակեցաւ ի մէնջ փոքր մի եւ յառաջագոյն, զի մարմին չէ հոգի, եւ ոչ շունչս, որ ի մեզ՝ մարմին. թողցուք զայլսն եւս։ 55 Արդ, մի բնութիւն բանին աստուածութիւնն է, իսկ միւս եւս ընկեր շարայարմարութեանն կամ զուգաճանապարհորդութեանն, ըստ անուղղայ բարբաջմանցն. զոր կանկնիցեն ի մարդկային բնութենէն՝ զհոգին առանց մտաց, թէ զանզգայ մարմինն, զի ոչ եւս գոյ հնար յաղագս այդորիկ ճոռոմել, մի բնութիւն ասելով զմարմին եւ զանձն։ 56 Ա՜յդ ուրեմն է բարձրայաւն աղանդոյն խորհուրդ, որ նման է գերեզմանաց բռելոց. զարտաքոյսն գեղեցիկ կեղծաւորելով, եւ ծածկեցելոյն փեռեկումն՝ ժահահոտութիւնս բղխէ։    57 Աստանաւր անբաւ տարակուսանաւք ունին պապանձել պարտութիւն յանձն առնլով, զի թէ երկուս ասել մաքառին, զմարդկային բնութիւնն աւցտեն, զրկելով զհոգին կամ զմարմինն ի փրկութենէ առաւղին։ 58 Նոյնպէս եւ զաստուածայինն հատանեն յերկուս, զդէմն ընդ մարդկային դիմին միաբանելով. եւ թէ հաւանեսցին միաւորութիւն խոստովանելով, որպէս էն իսկ, ոչ եւս այլ համարձակեսցին անդուռն բերանովք երկուց ձայնիւ ղաւղանջել։    59 Այլ մեք, ով սիրելիք, յամենայնի կարող եւ բաւական զանսկզբնական էութիւնն Հաւրն երկնաւորի փառաբանեսցուք, որ իմանալի ձեռաւքն զմարդոյն յարմարեաց կազմութիւն. այսինքն Որդւովն եւ Սուրբ Հոգւովն, ոչ որիշ զարտաքին կամ զներքին մարդն ասել, այլ՝ մի։ 60 Եւ զարարչական Բանին զտնաւրէնութիւնն ի Մարդ եւ յԱստուած, մի յաւակնեալ բաժանեսցուք ամպարշտաբար, զի խորհուրդ է անքննելի. գուցէ ի փայլակնացայտ մեծի աւուր գալստեանն, եւ զմեզ հրամայեսցէ կտրել ընդ մէջ յառաջ՝ հետեւող հրոյն ընդ կեղծաւորսն տալով բաժին. այլ ամբիծ հաւատով հանդիպել նմա, զի մի տէրութեամբ եւ կերպարանաւք ծագելոց է արեգակն ի բարձանց փառաւք Հաւր եւ ոչ երկուք։ 61 Եւ յայնժամ հայեսցին ի նա բաժանաւղքն, փախչել, եւ բաղձալ անսասանելի եւ անշուշտ հաւատոյ դաւանութեան ի մարգարէիցն տնկեալ, եւ յառաքելոցն զարդարեալ, պայծառացուցանելով ընդ տիեզերս, զի ընդ նոսին եւ պարգեւացն հասանիցեմք, շնորհաւք եւ
