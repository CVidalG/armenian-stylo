է խորհուրդ մերձաւոր տաւնիս, եւ տիեզերական ցնծութեան աւետիս հնչեցուցանէ ընդ երկնաւս։ 2 Գերապայծառ ճառագայթեալ լուսով ընդհանուր զուարթ եւ ուրախ տեսանեմ զհաւատացելոցդ գումարեալ զքրիստոսասէր դաս։ 3 Նորահրաշ արփիափայլ լուսով պսակեալ՝ զտաւնիս յաւէտ հրաշացուցանելով պայծառ ձայնիւ զխորհուրդ, զարթուցանել ընդ մեզ յերգս ցնծութեան եւ վերին Սիովնի զերկնագումար զպարս առաջին լուսաւոր բնութեանցն, յետ առաջնոյ անտի աղբերացեալ եւ ստորնայնոցն ÿի վերինն պարակցեալ խումբս՝ զբոլորից փրկութեան համախմբեալ վայելչացուցանեն բերկրալից ձայնիւ զհրաշս։    4 Տինա ոչ միայն մերասերիցս բնութիւն բանաւորական, եւ հրեշտակք առաջին միտք եւ մաքուրք զսորայս հռչակեն զգովեստ, այլ գրեթէ բոլոր իսկ տարերքս ÿի միասին լեալ ընծայաբերք զթագաւորականի աւուրս պայծառակերպեալ հրաշազարդեն զփառս։ 5 Վասն զի ամենայն ինչ ÿի բարի տաւնիս ÿի միասին գեղեցկապէս ընթացեալ զբնաւս զուարճացուցանեն զմիտս եւ զզգայութիւնս՝ հանդերձ որք բոլորովին են անշունչք եւ անշարժք։ 6 Զի այսաւր հրեշտակական դասուցն քաղցրանուագ հնչեն աւրհնութիւնք ÿի նոր Սիովն պայծառահրաշ լուսով. եւ փայլումն գալըստեանն Քրիստոսի զհաւատացելովքս փարի։ 7 Նորոգին երկինք, եւ երկիր լրիւ իւրով ըստ մարգարէին. եւ լերինք՝ ցնծան առաջի Տեառն։ 8 Ոչ միայն կենդանի լուսոյն ընդ բնութիւնս մեր խառնմամբ աղբերացեալ աւրհնութեամբքն տարամերժին անէծք, եւ տարերք յուսով յառաջինն նորոգին ÿի հրաշափառութիւն, այլ գրեթէ ակնյայտ երեւակի բոլորեցունց վայելչազարդեալ բնութիւնք՝ զպայծառս զայս պսակեն զտաւն իմանալի եւ զգալի փառաւք։ 9 Վասն զի այժմ երկին ÿի վերուստ պայծառազգեստեալ լուսաւորացն յինքն ընթացիւքն՝ ջահավառէ ընդհանուր վճիտ եւ մաքուր լուսով զգալի զայս աշխարհ։ 10 Իսկ հաւատոյն բարձրացեալ խորան ըստ երկնի կամարաձեւ զհաւատացեալսն շրջափակելով զանապական ճառագայթ անստուեր լուսոյն ÿի մտաւորս բացափայլեալ աշխարհ բանական բնութեանս։    11 Այժմ արեգակն պայծառալոյս ÿի վերինն հասեալ կամար հրաբորբոք ճառագայթիւք վանեալ զհիւսիսին՝ տապացուցանէ զընդ երկնաւս կենդանածնելով զբնութիւնս ÿի զգալիս։ 12 Իսկ արդարութեան արեգակն ÿի հաւատոյն փայլեալ խորան լուսաբորբոք հոգեռանդն բոցով հրավառէ զտարփողս անթիւրելի հաւատոցն յէականն կենդանագրեալ լուսոյ տիպ։ 13 Այժմ լուսնի բոլորակն լուսավառեալ եւ աստեղացն պար մաքրեալ՝ զերկնայինն հրաշազարդեն բուրաստան, եւ զտիեզերս ÿի գնացս իւրեանց ճառագայթակերպեն լուսարշաւ ընթացիւք։ 14 Իսկ հաւատոյն լուսոյ արմատ եւ աղբիւր՝ մերունակ բնութեանս պտուղ բարեբանեալն Աստուածածին ընդ երկնաւս մաքրապայծառ լուսով կուսութեանն ընդ իւր զհոգեւորսն վառեալ ղամբարաջահ զսրբոցն զաստեղս զԵրրորդութեանն լուսոյ ընդ տիեզերս շրջափայլէ հրաշապայծառ ճառագայթ։ 15 Այժմ ամպք՝ տէրունւոյն սպասահարկեալ հրամանի լուսատեսակք յարեւելից ոսկեճաճանչք ÿի հիւսիսոյ զմիմեամբք կուտակեալք ÿի միգագոյն թանձրութիւն փայլակնացայտ որոտընդոստ ցաւղաբուղխ առատութեամբք զտիեզերս արբուցանեն։ 16 Եւ աւդք շնչեալ մաքրագոյն բիւրակերպեան ծաղկացն եւ տնկոցն երփն երփն զարդուք զաշխարհս պայծառազարդեալ պսակեն։ 17 Իսկ առաքելական ամպոցն ընդ երկնաւս լեռնացեալ բազմաճառագայթ կուտակք աւետարանին ահեղ որոտմամբ փայլակնաբորբոք հրակէզ զաւրութեամբ մատռուակեն զանձրեւ ողորմութեան գիտութեանն Աստուծոյ ÿի հովիտս բնութեան մերոյ դաշտի. շնչեն եւ զհոգւոյն կենդանարար աւդ յաճումն բարութեան բանաւոր բուսոցն եւ առ ծաղիկս անուշահոտս վայելչացեալ քաղցր պտղովն։    18 Այժմ երկիր վայելչացեալ պայծառակերպի յերեւման Աստուծոյ մերոյ պսակակապ թագազարդեալ տերեւազգեստ ծաղկազարդեալ բուսովքն եւ տնկովքն. եւ լերինք ծաղկաւէտեալք ծիրաներփնեան ցնծացեալք ընդ առաջ Տեառն։ 19 Անուշահոտանան եւ խաղք, եւ կենդանեացն խայտան հոյլք ÿի դալարավայրս. եւ հաւուցն յաւդս վայելչացեալ երեւմամբք՝ քաղցրանուագ հնչեցուցանեն զանտառս։ 20 Իսկ բանաւոր բնութեանս մերոյ երկիր լուսազարդեալ ցնծայ ընդ գալուստ Աստուծոյ մշտազուարճ առաքինութեանն ծաղկաւք նորափետուր զարդարեալ, եւ ÿի հաւատոցն բարձրեալ լեառն՝ պսակէ զտաւնասէրս ոսկեճաճանչ իմաստութեանն ծաղկաւք. եւ աստուածգիտութեան բուրումն զերկնախոհիցն պարալնու անուշահոտութեամբ զիմաստս։ 21 Կթին եւ պտուղք անմահութեան զհոգւոյն քաղցրացուցանել սրբութեամբ զզգալիսն։ 22 Ցնծացեալ պարեն յեկեղեցւոյ շուշանազարդ վայրս եւ մաքրազգեստիցն լուսատեսակ դասք, եւ աւետարանին քաղցրաձայնութեան նուագք ըստ հաւուցն երգոց աւետալուր հնչեցուցանեն զընդհանուրս։     23 Այժմ ծով ÿի բարձրադէզ լեռնակուտակ ալեացն զնստեալ վրդովմանց՝ ասպարէզ ընթացից զկոհակս տէրունւոյն յարդարէ ոտից. եւ գետոց յարձակմունք առաւելապէս հոսին ընդ տիեզերս. եւ ականակիտ աղբերացն քաղցրաբուղխ ծաւալմամբ բնութիւնք զովացեալ հրճուին, զի Աստուած եկեալ է յաշխարհ։ 24 Իսկ իմաստութեան ծով բնութեան մերոյ խորք՝ ÿի չար հողմոց երբեմն անդնդապտոյտ փրփրատեսակ ալեացն հանդարտեալ, գնացս յինքեան անվայրափակ յարդարէ բնութեանս՝ հոգւոյն ÿի վերուստ շողացեալ ճառագայթիւք. եւ աւետարանական գետոցն տիեզերագնաց ծաւալմամբ ÿի կենաց աղբերէն զանմահութեանն անհատաբար յորդեալ հոսանս զտիեզերս պարածածկեն ողջունիւ խաղաղութեամբ յերեւել Աստուծոյ մերոյ։    25 Այսոքիւք տաւնիս նկարագրի պայծառակերպութիւն այսպիսեաւք հրաշիւք։ 26 Արդարեւ սա է աւր ցնծութեան եւ ուրախութեան զոր արար Տէր. սա է աւր ծագման մեծի փառացն Աստուծոյ, որ է նոր արքայութեան սկիզբն եւ տեսչութիւն էական լուսոյն եւ նկարագիր յուսոյն յաւիտենից։ 27 Սա է աւր սուրբ միաշաբաթուն պսակեալ լուսով տէրունական յարութեամբն. յորում եւ էջք Հոգւոյն Սրբոյ ÿի վերնատունն պայծառ լուսով շրջափայլեալ՝ զերկնախմբիցն ÿի նմա հրաշազարդեաց զդասս։     28 Սա ÿի Մովսիսէ անուանեցաւ աւր մի յորում սկզբնածինն լուծաւ խաւար, եւ ÿի լոյս ամենայն վայելչացաւ բնութիւնս։ 29 Իսկ ÿի մարգարէէն աւր ութերորդ, որ նշանակէ նոյն մի. եւ ութերեակ վասն միոյ մեծ եւ աներեկ աւուրն, որ ÿի միում ճշմարիտ աշխարհին յարատեւակ աստուածային լուսով մաքրեցելոցն զանլուծանելին ընձեռէ կեանս. որոյ ճառագայթք պայծառութեան անդ լուսաւորելոցն յայսմ փայլէ աւուրս, յորում փրկիչն մերունակ կերպիւ զաստուածութեանն ÿի Թափաւր ճառագայթեաց զլոյս՝ փրկական փայլման լուսոյն առ հաւատչեայ ÿի յարութեան սրբոց աստուածանմանակերպութեամբ, զոր եւ տաւնէ եկեղեցի հրաշապայծառ ցնծութեամբ։    30 Սա է տաւն տաղաւարահարացն աւրինադրեալ Մովսէսի յետ ելիցն Իսրայէլի ÿի հնգետասան աւուր ամսոյն եաւթներորդի. յորում զեաւթներեակն ÿի տաղաւարս բնակեալ համախմբեալ տաւնէր Իսրայէլ, բարձեալ ունելով կողերս արմաւենեաց եւ ձողս հագնեաց եւ զոստս եւ զուռս սաղարթազուարճս առ ÿի յիշատակ խորանաչու ընթացիցն որ ÿի Սինա եւ ընծայաբերութեան գոհութիւն աստուածական երախտեացն, զոր բարձիւ կրեցին, յաւետեացն աստուածակերպ բուրաստանատեսակ ժամանեալ աշխարհ։ 31 Այս առ նոսա ստուերական աւրինադրեալ տաւն, որ զաւետարանին նախագրէր զճշմարտութիւն. քանզի այն աստուածգիտութեան լեառն, յորում է Աստուած որ զամենայն նիւթականս ÿի նմա հրով բոցակիզէ։ 32 Ի նմա զփառսն Տեառն իբրեւ ընդ հայելի տեսեալ՝ ÿի նոյն լուսոյ կենդանակերպիմք, եւ փառաց ÿի փառս ճառագայթեալ աւրինադրիմք զկենցաղոյս լքանել նիւթական շինուած, եւ յաննիւթն փութացեալ յարկ՝ զքակտելիս զայս մարմնոյ անդամս առաքինութեան յարդարել խորան։ 33 Ի սմա որպէս ÿի տաղաւարս տաւնեմք յեաւթներորդում ամսեանս ըստ յաւիտենիս այսորիկ եաւթնեակ շրջագայութեամբ ÿի հնգետասաներորդում աւուր ամսոյն՝ զգայութեանցս երիցս ընկալեալ մաքրութիւն. յորս գայթակղելն եւ ընդ որս պատերազմ ÿի ներքս ընդունեալ զխայթոց մեղացն։ 34 Եւ տաւնեմք զեաւթներեակն ÿի տաղաւարս, կողերս բարձեալ արմաւենեաց, եւ զոստս եւ զուռս կանաչազգեացն. որ է զբոլորս զայս եաւթնեակ շրջագայութեան կենաց հոլովս արտաքսելով այսմ աշխարհի՝ կեալ ըստ Աստուծոյ հոգւով, եւ անմարմնական առաքինութեանց ոստոցն աճմամբ առ ամբարձումն հրեշտակաց բարձրածայրեալ որպէս արմաւենիս ÿի գաւիթս Աստուծոյ ծաղկիմք առ վերինն հրաւիրեալ երկիր աւետեաց ÿի Սիովն քաղաք Աստուծոյ ÿի մայր կենդանեաց եւ ÿի բանակետղ անմահից ÿի լուսոյն առագաստ յանքակտելի խորան, ուր տաւնողացն է հնչումն յարազուարճ խնդրութեամբ։    35 Այսոքիկ մեզ յաւրինակ կանխաւ տպաւորեալք. զոր ինքեամբ ելից՝ որ գրոյն է լուծիչ Քրիստոս եւ կատարիչ խորհրդոյն, ÿի սմին աւուր տաւնի աշակերտացն ÿի լերինն անմերձենալի զաստուածութեանն նազելի փառացն յայտնեաց զծայրագոյն լուսոյ ծագումն ÿի կենդանարար աղբերացեալ մարմնոյն փայլակնատարած ճառագայթարձակ փայլմամբ։ 36 Եւ էր տեսանել զԹափաւր նոր զոմն Սինա լեառն սուրբ եւ վայր աստուածաբնակ լուսեղէն, վառեալ հրով եւ ամպով. այլ եւ երկին նոր ÿի վերայ երկրի բարձրացեալ աստուածութեանն ճառագայթիւք՝ պայծառազգեստ պսակեալ արփիափայլ հրաշիւք։ 37 Ի նմա եւ զաւրք հրեշտակաց բիւրաւոր բանակք սերովբէից գումարեալք. որք ÿի ներքոյ տէրունւոյն ոտից սոսկացեալ՝ հրափայլ թեւաւքն պարփակեալ զարքայական աթոռովն։ 38 Զի ուրանաւր վերանստեալն, անդ եւ աթոռ նորա, անդ եւ պարք բարերջանկացն՝ պայծառ եւ մեծ առաքելոցն եւ մարգարէից լուսազգեաց դասք հպաւորութեամբ անմերձենալի լուսոյն աստուածակերպեալք, բացաւ երեսաւք առ Աստուած հայելով զմերս լուսակերպելով մթացեալ խաւարային բնութեանս զսեաւ աղջամղջին տեսակ ինքեամբ միացուցեալ առ Հայր, ըստ որում եւ ասաց. Որպէս դու Հայր յիս եւ ես ÿի քեզ, զի եւ նոքա ÿի մեզ իցեն։ 50 Վասն զի ելանելով ÿի լեառնն՝ ÿի վեր տանէր զմեզ ինքեամբ ÿի հայրական աթոռն. եւ զմերասեր մարմնոյն արեգակնափայլ լուսազարդելով զդէմս՝ զմերս աստուածակերպէր խոնարհութեան զմարմին նմանութեամբ մարմնոյ փառաց իւրոց. ըստ որում եւ կամարաձեւ շրջափակմամբ լուսեղէն ամպոյն՝ զմերս առագաստէր ÿի սրբութիւն սրբոցն ընդ ինքեան զբնութիւնս ÿի ներքոյ վարագուրին, որպէս ասացն՝ թէ Ուր եսն եմ, անդ եւ պաշտաւնեայն իմ եղիցի։ 51 Եւ հայրականն ձայն զինքն ÿի վերուստ վկայեալ որդի սիրելի, զորդեգրութեան մեր զառհաւատչեայ տայր աւետիս հրաւիրելոցն ÿի լոյս. նա եւ զառաքելոցն եւ զմարգարէիցն խմբիւք զծայրագոյնն իւր ճառագայթելով լուսափառս՝ զբոլորիցն նշանակէր զվերնոց եւ զներքնոց կենդանեաց եւ մեռելոց տեսանել զինքն ÿի վերջին աւուրն ընդ ամպս եկեալ զաւրութեամբ եւ փառաւք՝ Աստուածն ÿի մէջ աստուածացելոց, զանդերանութեանցն աստ մաքրեցելոցն տալով զփառս. որ ÿի մեծի ահաւոր աւուրն լուսահրաշ փառաւք տաւնս այս պսակեալ արեգակնակերպ փայլէ ընդ երկնաւս յինքեան ունելով զերեքպայծառեան լուսոյն եզաբուն ճառագայթ, զոր միածինն այսաւր ÿի թափաւրական լերինն զԵրրորդութեանն բոլոր յինքն փայլեաց զլոյս, ճառագայթելով զարարածս. որով երկինք ցնծացան եւ երկիրս պայծառացաւ, վերնայարկք երկնից բացան եւ լոյս ÿի ստորինս հոսեաց խորան. հայրականն ընդ երկնաւս հնչեաց ձայն. տիեզերք որդեգրեցան. ճառագայթ հոգւոյն ծագեաց. երկրածինքս լուսազգեցան. հրեշտակացն խումբ յերկրի պարեաց. հողեղինացս դասք մտին ընդ դրունս երկնից. աւրհնութիւնք յերկնից յերկիր հնչեաց. երկիր երկնանման զարդարեցաւ. դրախտին դրունք բացան, արտաքսեալն անդրէն պարեաց ÿի նմա. հոտ ծառոյն կենաց բուրեաց. երկիր զանմահութեան բուսոյց ծաղիկս. կենաց լոյսն ծագեաց. մահու ստուերն փախեաւ. արդարութիւն թագաւորեաց. իշխանութիւն մեղացն լուծաւ. աւետարանական փողն գոչեաց, տիեզերք ÿի քնոյ զարթեան, զԱստուած մարմնով եկեալ տեսին, եւ ընդ նմա աստուածացան։     52 Այսոցիկ մեզ բիւր բարեաց փոխանակ այս տաւն մեծ եւ պաշտելի։ 53 Եւ եւս յոյժ վայելուչ է անուն տաւնիս զոր կոչէ եկեղեցի. զի վարդավառ ասի, որ է արդ աւար, եւ կոչի վարժարան։ 54 Այլ վասն զի յոյնն զսա պայծառակերպութեան եւ պսակաց կոչէ տաւն, իսկ ազգ հայաստանեայց վարդավառ, նոյն եւ մի զսորայս ձայնէ հրաշափառ պայծառութիւն, զոր կանխաւ բանն տպաւորեաց։ 55 Ի սմա բովանդակ երկիր ծաղկաւէտեալ, եւ բնութիւնս մեր յաւիտենական փթթեալ ծաղկաւք, եւ ճառագայթարձակ փայլմամբ փառացն Քրիստոսի. եկեղեցիք պսակեալ պայծառազգեստ վայելչութեամբ եւ առաքինութեանն լուսահրաշ վարդիւք եւ շուշանաւք՝ տաւնասիրացն ոսկեվարսեալ վայելչանան գլուխք. ըստ այսմ յիրաւի վարդավառ զսա ձայնէ եկեղեցի, եւ պայծառակերպութեան եւ պսակաց տաւն։ 56 Իսկ արդ աւարն եւ վարժարանն յայլ բարձրագոյն տրամադրի խորհուրդ, զոր տրամախոհիցն ÿի հոգւոյն է գիտել։ 57 Քանզի այն ինչ ծագեալ ÿի մեզ յարութեանն Քրիստոսի լոյս ÿի գլուխ սրբոյ յիսնեկիս, հոգւոյն յառաքեալսն ÿի վերնատանն զեղան շնորհք. յորում աւուր աստուածագիծ աւրինացն տառ ÿի Սինա ընձեռեալ Մովսէսի զոր տաւնէր Իսրայէլ ամ յամէ՝ պենտեկոստէ զսա առձայնեալ, ÿի նմին աւուր եւ հոգւոյն յառաքեալսն զիւրն կենդանագրեալ շնորհս՝ տեղի տայր գիրն, եւ աւետարանին բազմաձայն լեզուաւք ÿի մի հնչեցեալ ձայն զլսողացն աշակերտէր հինգհազարեան խմբիցն հոյլս՝ նոր Իսրայէլ զնոսա գրելով. առաջին քրտունք երկնաւոր մշակին եւ ծաղիկք առաքելական ոստոցն, որք զգրոյն ÿի բաց մերկացեալ զպատրուակ՝ հոգւոյն զգեստաւորեցան լուսով, աւետարանին հնազանդեալք աւրինադրութեանց՝ զոր առաքելական հրամանն զնոսա յերկրորդս յայս փակեալ յիսներեակ՝ որպէս ÿի վարժարանի վարժէր զնոսա ÿի հրահանգս վարժից հոգեւոր կրթութեանց ըստ իսրայէլեան երբեմն շրջացանական մաքրութեանն յերեւման Աստուծոյ։ 58 Սայապէս եւ սոցա սրբութեամբ աղաւթից մաքրութեան աւազանին լուսով որդեգրութեանն արժանանալ շնորհի, ըստ այսմ վարժարան զսա կոչել ոչ ինչ եպերելի է։ 59 Իսկ արդ աւարն քանզի յանկման առաքելական աւրինադրութեանցն ÿի սմին յայսմ աւուր տաւնի տաղաւարահարաց առաջին յԵրուսաղէմ զեկեղեցւոյ կանգնեցին խորան, ÿի նմա զնորընծայս լուացմամբ աւազանին որդեգրեալ հաւրն լուսոյ՝ փոփոխելով զհնութիւն ահարոնեան ծերացեալ պաշտամանն ÿի նորոգութիւն Քրիստոսի քահանայութեանն, եւ զկերպ գրոյն ÿի հոգւոյն շնորհս, եւ զթլփատութիւն մարմնոյն ÿի մտաց մաքրութիւն, եւ զառաւրեայ լուացմունս ÿի մշտնջենաւոր աւազանին տաղաւարեալ՝ ÿի հաւատոյն աւրինադրեցին նորածնելոցն բարձրանալ ÿի լեառն, Երրորդութեանն պսակեալ լուսով, ծիրանիս արկեալ զհոգւոյն ճառագայթ, եւ երկնածաղիկ առաքինութեանն լուսասաղարթ ոստովք քաջուղէշ եւ յարափթիթ մէջ իւրեանց զտիեզերս ամենայն կալ իշխանս ÿի ներքոյ երկնից ըստ կանխասացութեան Հոգւոյն Սրբոյ Ի գումարել երկնաւորին զթագաւորս, առնուլ յաւարի զընդհանուրս։ 63 Ուստի առ միմեանս գոչէին միախումբ երկնաւոր նահատակացն աշխարհագոչ փողովք ասելով. Արդ աւար է, ելէք ընդ տիեզերս գերել զգերութիւն, առնուլ աւար, բաշխել պարգեւս եւ տալ որդւոց մարդկան. որպէս գրէ խորհրդապետն այսոցիկ Պաւղոս, եթէ Գերեմք զամենայն միտս ÿի հնազանդութիւնն Քրիստոսի։ 64 Աստանաւր ցուցաւ թէ արդ աւար է տաւնս, զոր վարդավառ զսա ձայնէ եկեղեցի, առաքինութեան ÿի սմա լուսերանգ եւ յուսահոտ բոցատերեւ վարդիւք հարսնազգեստ պսակեալ եւ հրճուեալ։ 65 Սա առաջին յեկեղեցւոջ աւրինադրեցաւ յառաքելոցն տաւն աւետաբեր, որ ոչ մտանէ ÿի շրջանս ընթերցուածոցն ընդ այլսն յորս աստուածային տնաւրէնութեանն վերաձայնաբար տաւնի խորհուրդ մարգարէիւք նախապատմեալ. զի ոչ զանցելոցն որպէս զնոսա, այլ զհանդերձեալ յաւիտենին յինքն բարձեալ ունի մշտապայծառ փառաց գերակայութիւն յարազուարճ ցնծութեամբ. որ եւ ոչ մարգարէքն ծանեան զայնս զոր անդն պատրաստեաց Աստուած սիրողաց ինքեան. զի սա ըստ ութերեակ աւուրն որպէս առաջինն այն եւ վերջին մշտնջենահոս լուսով անթառամ պսակեալ տաւն յաւէտահրաշ եւ ճառագայթաւէտ, սկիզբն առեալ ÿի դատողական ատենէն, եւ տեւէ մինչեւ յանսպառ մշտանորոգ յաւիտեանսն զտաւնասիրացն բացախմբից միացեալ պար յիւրն լուսաթաղանթ առագաստեալ ծոց, զանզրաւական ուրախութեանցն տրոհելով աստ մաքրեցելոցն յարատեւողութիւն ընդ էին՝ աստուածացեալ եղելոց, զոր ոչ ոք կարէ ասել զայն զի եղիցի Աստուած ամենայն յամենայնի։    66 Իսկ մեր զաստուածահրաշ լուսաւորութեանցս տաւնախմբեալ զաւր՝ կնքեսցուք զբանս. պայծառազգեաց պսակեալ լուսով փառացն Քրիստոսի ընդ սրբոց առաքելոց եւ մարգարէից պարս՝ արարչական սպասաւորեսցուք տնաւրէնութեան, զլուծանելի մարմնոյս տաղաւար՝ մաքրութեան առնել խորան. յորում զմի եւ զնոյն ինքն զՔրիստոս ընդունելով, զԵրրորդութեանն ÿի սիրտս մեր անպարագիր ընկալցուք լուսաւորութիւն. եւ սապէս ընդ լուսերամիցն յափշտակեսցուք ամպովք ընդ առաջ Տեառն յաւդս նոյնատիպ աստուածացեալ կերպիւ՝ զուարթ եւ պայծառ կալ յառաջագահ դասուն, զորդեգրութեանն առեալ պարգեւս ընդ թագաւորեալսն ÿի լոյս ÿի նոյն ինքն ÿի Քրիստոս Յիսուս ÿի Տէր մեր. որում փառք յաւիտեանս յաւիտենից. ամէն։    ՊԱՏԱՍԽԱՆԻ ԹՂԹՈՅՆ ՍԱՀԱԿԱՅՅորում եւ բանք մարգարէականք եւ վասն փոխման Աստուածածնին թէ որպէս եղեւ կամ որպէս շնորհեցաւ անարատ տիպ պատկերին ի փայտն եւ թէ յոյր ձեռն եւ զինչ եղեւ պատճառք խնդրոյն կամ թէ յումմէ բերաւ ի Հայս        1 Զյորդութիւն առատութեան աստուածային հոգւոյն շնորհաց, որ ՚ի վերայ քո անդուլ մտացդ, որ վասն գրոց ծանաւթութեան եւ սիրոյ առ Աստուած, յառաջ քան զտեսութիւն ՚ի ձեռն համբաւուց լուեալ իմ զխոհականութիւն մեծի հզաւրիդ ծանեայ, եւ այժմ ՚ի ձեռն արժանափայլ խնդրոյս յառաջ քան զնկարագիր պատկերիդ եւ զտիպ մարմնոյդ, տեսեալ ՚ի վրէժխնդրութիւն հոգւոյդ, գիտացի որ ՚ի վերին շնորհացն է այս։ 2 Վասն որոյ եւ ինքն բանն Աստուած ՚ի ստեղծանելն զմարդն բանիւ զարդարեաց առ ՚ի յաւրհնել եւ ՚ի ճանաչել զերախտագործն, յորս ասէ. Աւրհնեա անձն իմ, զՏէր, եւ մի մոռանար զգործս նորա։ 3 Յայսոսիկ ուրախ լինի տէրն, եւ երանէ զայնոսիկ որ յաւրէնս տեառն խորհեսցի ՚ի տուէ եւ ՚ի գիշերի. զի եղիցի նա որպէս ծառ ՚ի գնացս ջուրց անթառամ, որ զպտուղ իւր ՚ի ժամու տացէ։ 4 Զի թէպէտ եւ երկիրս զիւրն սիրէ, սակայն ոչ է գովելի, զի եղծանելի է. զի Աստուած ստեղծ զմարդն յանմահութիւն որպէս ասացն, այլ ստեղծականաւն կերակրեալ մահացաւ։ 5 Վասն որոյ զբանին վառ եւ բորբոք պահելով զկայծակն անմահանամ։ 6 Ասացից քեզ ՚ի բազմաց զերկուց քերց զիրս միայն, զի երկոքեանն ջերմութեամբ զտէրն ընկալան։ 7 Մարթա կերակրելովն, այլ գովելի է Մարիամ, զի ասացաւ յաստուածեղէն բերանոյն. Աստ սակաւ ինչ պիտոյ է. Մարիամ մասն բարի ընտրեաց, որ ոչ բարձցի ՚ի նմանէ։    8 Յայսցանէ գիտացի լեալ զքեզ, մեծ իշխանդ Վասպուրականի, որ թէպէտ եւ ՚ի նախարարս ես թուեալ, սակայն ՚ի տանէ թագաւորազուն, եւ ազգաւ ես քաջ եւ նախնական, վախճանելն Հեղեայ, յետ մնալոյ իւրոյ անզաւակ, հարկեալ Յակոբ ըստ աւրինացն, առնու զկին եղբաւր իւրոյ յարուցանել նմա զաւակ։ 58 Եւ այսպէս ծնաւ Յակոբ ՚ի կնոջէն Հեղեայ զՅովսէփ եւ զԿղէովպաս, ծնունդք ըստ բնութեանն Յակոբայ, եւ որդիք Հեղեայ ըստ աւրինացն։ 59 Իսկ Յովակիմ հայր Մարիամու, եւ նորին հայր Պանթեռ, եղբայր Մեղքեայ, էին որդիք Ղեւեայ, ազգակիցք Յակոբայ, որպէս ցուցանեն եբրայեցւոցն աւանդութիւնքն։ 60 Որպէս զի գոլ Մարիամու դուստր Յովակիմայ, թոռն Պանթեռայ եղբաւր Մեղքեայ, որդւոյ Ղեւեայ, որ առ զկինն Մատթանայ։ 61 Իսկ կին Հեղեայ, տուեալն Յակոբայ ծնաւ զՅովսէփու, որպէս յառաջագոյն ասացաք։ 62 Որպէս զի գոլ Մարիամու, որ եղեւն Յովսէփ այր, ըստ երկուց նահապետութեանցն, Նաթանայ եւ Սաղմոնի, յազգէ Դաւթի ըստ հաւր, իսկ ըստ մաւր Հեղեայ, եղբաւր նորին հաւրն Յովակիմայ, եւ ըստ քահանայութեանն ազգակից Եղիսաբեթի։    63 Եւ ինքն սուրբ կոյսն սքանչելի վարուք սրբութեամբ առաւելեալ քան զամենայն ոք, ըստ բանին Սողոմոնի, Բազում դստերք ստացան զմեծութիւն, բայց դու առաւել եղեր, որպէս աւետարանեցաւ ՚ի Նազարեթ ՚ի Գաբրիէլէ, միահամուռն ամենայն ամք կենաց նորա վաթսուն ամ. ՚ի տասն եւ հինգ ամին ծնաւ զփրկիչն աշխարհի, եւ երեսուն եւ երեք ամ մինչեւ ՚ի խաչելութիւն փրկչին։ 64 Իսկ յետ համբառնալոյ յերկինս փրկչին երկոտասան ամ, որ լինի ՚ի սկզբանէ կենաց նորա մինչեւ ՚ի հրաշալի փոխումն վաթսուն ամ։     65 Սուրբ կոյսն եւ Եղիսաբեթ եւ Սողոմէ, երեք քեռ երեք դստերք էին սոքա. եւ Մարիամ Կղէովպայ քոյր ասի Աստուածածնին, զոր ասէ՝ կային առ խաչին Յիսուսի։ 66 Այլ եւ Յովսէփ այր Մարիամու ունէր որդիս չորս, զՅակոբոս, զՍիմոն, զՅուդա, զՅովսէս. ունէր եւ դստերս երիս, զՄարթա, զԵսթեր, զՍողոմէ՝ որ էր մայր Յակոբայ եւ Յոհաննու եւ կին Զեբեդեայ։ 67 Եւ Յոհաննէս կոչի քեռորդի փրկչին. զի որպէս որդիքն Յովսէփու եղբարք ասին փրկչին, այսպէս եւ դստերքն քոյրք։ 68 Յովսէփ քառասունամեայ էր մինչ կոյսն յանձնեցաւ նմա, եւ յերեսուն եւ մի ամի կենարարին հանգեաւ ՚ի Նազարեթ ՚ի տան իւրում. եւ սպասաւորէր սրբոյ կուսին Յակոբ արդարն ՚ի Յովսէփայ յանձնեալ, զի եւ առ Յովսէփաւ նա հոգայր ծառայապէս, հեզն եւ արդարն, մինչեւ ՚ի Քրիստոսէ Յոհաննու յանձնեցաւ։    69 Եւ սուրբ կոյսն բազում չարչարանս կրեաց եւ վտանգ ՚ի Կայիափայ, եւ հալածանս ՚ի Հրէիցն։ 70 Զի ոչ տային նմա գնալ ՚ի գերեզմանն կալ յաղաւթս, կամ ՚ի Գեթսեմանի յայտնապէս, կամ յայլ տեղիս տնաւրինականս փրկչին, բայց միայն գաղտնի։ 71 Եւ հալածեալ զեղբայր տեառն զՅակոբ, երթեալ հուր հարին տանն յորում կայր սրբուհին, եւ շրջապատեալ հուրն զՀրեայսն, կիզեաց արս եաւթն, եւ զԿայիափա եւ զԱննա եւ զՂեւի քահանայապետք, ոչ ինչ վնասեալք տանն։ 72 Եւ երկուցեալք, այլ ոչ եւս իշխէին համարձակել ՚ի սուրբ անձնն։ 73 Յաւուր մկրտութեանն տեառն, խաժամուժ ամբոխ Հրէիցն գնացեալ հրոսակք ՚ի Յորդանան ըմբռնել զՏիրամայրն եւ զՅովհաննէս։ 74 Իսկ հրեշտակ Աստուծոյ եդ զնոսա յայնկոյս գետոյն. եւ Հրեայքն անցանել կամեցեալ, արս քսան ՚ի նոցանէ կորոյս Յորդանան։ 75 Եւ յաւուր համբարձմանն տեառն, ՚ի լեառն ձիթենեաց յախուռն քարամբք հասեալ Հրեայքն ՚ի վերայ սրբուհւոյն սատակեցին յինքեանց արս յիսուն քարավրէժս։ 76 Այլ զի բազում հալածանս գրելով զՏիրամաւրն, ճառս ձանձրասցի յընթեռնուլ, այսուհետեւ ՚ի խնդիր սրբոյ պատկերին դարձայց։    77 Հրաւիրեալ սրբուհին ՚ի փոխումն առ որդին իւր, յայտնեալ աւետարանչին զփայտն կիւպարիս, յորմէ կենսաբեր խաչն, մեծաւ թախանձմամբ կերպաձեւէ զպատկեր Տիրամաւրն առ ՚ի մնալ նշխար կենաց մերոց, առ անձուկ սրտին իւրոյ յոյժ բաղձացեալ աւետարանիչն։ 78 Քանզի յետ համբառնալոյ փրկչին մերոյ, նովաւ մխիթարէին աշակերտքն, զսուրբ կոյսն տեսանելով որպէս զորդին իւր. նա եւ Պետրոս եւ Պաւղոս բազում անգամ ելանէին յԵրուսաղէմ տեսանել զսուրբ տեղիսն, եւ ՚ի տեսութիւն իսկուհւոյն։    79 Արդ ՚ի ժողովել աշակերտացն համագումար միաբանութեամբ ՚ի յղարկել զաստուածամայրն իւրեանց, ըստ բանին տեառն, եթէ ուր մարմին է՝ անդ եւ արծուիք ժողովեսցին. որք թէպէտ եւ ՚ի ժամ խաչելութեանն ցրուեցան, այլ ՚ի ժամ յարութեանն ժողովեցան. քանզի ասաց տէրն, քակեցէք զտաճարս զայս եւ ես յերիս աւուրս շինեցից զսա։ 80 Զոր մարգարէն ասէ, շինէ զԵրուսաղէմ տէր, եւ զցրուեալսն Իսրայէլի ժողովէ։ 81 Խորհեցեալ այնուհետեւ եւ Յոհաննու ընդ այլ առաքեալսն աղաչել զՏիրամայրն իւրեանց, դնել զփայտեղէն անաւթն ՚ի վերայ աստուածատիպ պատկերին իւրոյ, եւ աւրհնել զնա, եւ խնդրել ՚ի տեառնէ բարի առնել երկրի ՚ի փոխմանն իւրում. քանզի ախտ բորոտութեան մեծապէս ցասմամբ եռայր յերկրի, բայց յԵրուսաղէմ առաւել եւս։ 82 Երթեալ Պաւղոսի եւ Յոհաննու, հանդերձ միաբան գնդիւն աղաչել զՏիրամայրն, ասէ Պաւղոս ցամենաւրհնեալ սուրբ կոյսն. Անարատ առագաստ Աստուծոյ Բանին, աղաչեմք զքեզ, ընկալ զփայտեղէն պատկերս զայս զնկարեալս ՚ի Յոհաննէ ՚ի սուրբ ձեռս քո, եւ աւրհնեա զսա, եւ տուր զսա նշան կենաց աշխարհի. եւ աղաչեա զորդին քո եւ զԱստուածն մեր, բառնալ ՚ի ձեռն պատկերիս զբորոտութիւն յորդւոց ՚ի մարդկանէ, որք ապաւինին ՚ի քո բարեխաւսութիւնդ։    83 Պատասխանի ետ ամենաւրհնեալ սուրբ կոյսն եւ ասէ ցնոսա. Եւ զիարդ հարկեցուցանէք զիս զաղախին տեառն առնել յանուն իմ իրս որ չէ արժան։ 84 Դուք աշակերտք գոլով արարչին մերոյ Քրիստոսի, ձեզ շնորհեաց իշխանութիւն հալածել զդեւս, եւ բժշկել զախտս եւ զհիւանդութիւնս, կոխել զաներեւոյթ գազանն, եւ բառնալ յաշխարհէ զկռապաշտութիւն։ 85 Դուք վասն որոյ իրիք եւ կամիք, աղաչեցէք զտէր ձեր եւ զԱստուածն, եւ լինիցի ձեզ. զի թէ զերկինս փակել իշխանութիւն առէք յերկրի, եւ կարգեցայք իշխանք ընդ ոլորտս տիեզերաց, եւ զձեզ բարեկամս իւր կոչեաց տէր, եւ նշանք հրաշալիք վասն ձեր լինին, եւ ուր ոտք ձեր հասանեն ոչ ինչ անհար լինիցի ձեզ։ 86 Այլ զիս աղաւթիւք ձերովք յղարկեցէք, եւ միաբանեալ ընդ իս սիրով խաղաղութեամբ՝ հասուցէք յանխռով նաւահանգիստն։ 87 Ասէ Պաւղոս. Կոյս սուրբ եւ մայր Աստուծոյ, մեք ծառայք անպիտանք եմք արարչին աշխարհի, որ հաճեցաւն ՚ի քէն մարմնանալ. արժանաւորեալք եւ մեք յորդւոյն քումմէ եւ յԱստուծոյն մերոյ առաքելական շնորհիս այսմիկ զոր ունիմք. իսկ դու տիրամայր գոլով մեր, եւ մեք հրովարտակք ընդ աշխարհս. դու հարսն Երրորդութեանն, եւ մեք փողք հնչեցեալք զանճառ ծնունդ որ ՚ի քէն. աղաչեմք զքեզ, արա մեզ նշան զպատկերս զայս, որպէս եւ միածին որդին քո, որ զփայտ մահուն իւրոյ անմահութեան գործարան ետ աշխարհի, եւ գանձ կենաց, հալածիչ դիւաց։ 88 Եւ այժմ շնորհեա, զի սա լիցի բժիշկ վիրավորեցելոց, եւ նշխար հաւատացելոց. զի տեսանելով զսա, որք փափագեն տեսանել զքեզ, լցեալ եղիցին խնդրուածք իւրեանց։    89 Յայնժամ առեալ ամենաւրհնեալ սրբուհւոյն զպատկերն նկարեալ ՚ի սուրբ ձեռս իւր, տարածեաց զաստուածընկալ ձեռս իւր յերկինս եւ ասէ. Տէր Աստուած ամենակալ, հայր տեառն մերոյ Յիսուսի Քրիստոսի, բարերար եւ բազումողորմ, որ առաքեցեր զմիածին որդիդ քո վասն կենաց աշխարհի, մարմնանալ յանարժան աղախնոյս քո, որ կատարեաց ՚ի մարմնի իւրում զգրեալսն աւրինաց եւ մարգարէից, եւ դարձեալ համբարձաւ առ քեզ ՚ի հայրենի քո փառսդ, աղաչեմ զքեզ այժմ եւ խնդրեմ ՚ի քէն, բարձ տէր զախտ բորոտութեան ՚ի մարդկանէ ՚ի ձեռն այսր նկարեալ փայտեղէն պատկերիս, զզաւրութիւն երրորդութեանդ տպաւորելով ՚ի սմա ՚ի ձեռն իմոյ աղաչանացս եւ խնդրոյս, որ բարձեր զյանցանս առաջին մարդոյն ՚ի ձեռն միածնի որդւոյ քո տեառնդ մերոյ եւ փրկչիդ Յիսուսի Քրիստոսի, որ կատարեաց ՚ի մարմնի իւրում զկամս առաքողիդ հաւր, կատարեա եւ այժմ զխնդիր արանց ընտրելոցս այսոցիկ՝ որ բանիւ քո առաքեցան ՚ի փրկութիւն աշխարհի, աւետարանել զԵրրորդութիւնդ սուրբ. զի քեզ լիցին երկրպագուք եւ փառաբանիչք հաւր եւ որդւոյ եւ սուրբ հոգւոյդ յաւիտեանս. ամէն։    90 Եւ ասացեալ առաքելոցն զամէնն, նոյն ժամայն կամար լուսեղէն եկաց ՚ի վերայ սրբուհւոյն, եւ լոյս ամպեղէն խաչանման կաթեաց ՚ի սուրբ պատկերն։ 91 Եւ եդեալ սրբուհւոյն զսուրբ պատկերն ՚ի վերայ երեսաց իւրոց, եւ թացեալ յորդառատ արտասուաւքն զոր հեղոյր առաջի տեառն։ 92 Եւ առեալ առաքելոցն Պետրոսի եւ Պաւղոսի զպատկերն ՚ի ձեռաց սրբուհւոյն, եւ ամենեքեան երկրպագեցին։    93 Իսկ յորժամ աւանդեաց զհոգին իւր սուրբ անարատ կոյսն ՚ի ձեռս տեառն, ամբոխ Հրէիցն յուզէր. եւ հասեալք ՚ի վերայ առաքելոցն, կամելով յափշտակել զմարմին կուսին, եւ հարան ամենեքեան կուրութեամբ։ 94 Իսկ այր ոմն անուն Սոփոնիա բուռն եհար, քանզի առաքելոցն բարձեալ տանէին ՚ի հանգստեան սուրբ տեղին ՚ի կոչեցեալն Գեթսեմանի, եւ մնաց ձեռք Սոփոնիայ կախեալ յարմկացն զմահճաց. լայր եւ աղաչէր այրն, եղուկ զինքն կոչելով մինչեւ ՚ի գերեզմանն։ 95 Ասէ Պետրոս. Ով մոլորեալ Հրեայք, մինչեւ ցերբ կայք մոլորեալ եւ յամառեալ, զիարդ ոչ զղջանայք յանզեղջ սիրտս ձեր, այլ կայք յետս հեստեալք որպէս հայրն ձեր սատանայ։ 96 Եւ լալիւն մեծ էր Հրէիցն։ 97 Ասէ Պաւղոս. Դարձարուք առ տէր, եւ խոստովանեցէք Աստուածածին զսուրբ կոյսն եւ գթասցի ՚ի ձեզ։ 98 Աղաղակեաց ամենայն ամբոխն եւ ասեն. Խոստովանիմք զՔրիստոս որդի Աստուծոյ հաւր, որ էառ մարմին անճառապէս ՚ի սուրբ կուսէդ, զոր մեքն խաչեցաք անգիտութեամբ, հակառակ Աստուծոյ կարծելով. միայն գթասցի ՚ի մեզ։ 99 Եւ առեալ զձեռանէ նորա Պետրոսի, եբեր մերձեցոյց ՚ի հատուածս ձեռացն, եւ միացեալ յարեցան իբրեւ զառաջինն։ 100 Եւ լուսաւորեցան խաւարեալքն, եւ մկրտեցան Հրեայք եւ հեթանոսք յոյժ յոյժ բազմութեամբ. եւ բորոտք որ էին ՚ի քաղաքին, գային առ առաքեալսն եւ աւծանէին իւղով, եւ տեառնագրէին սուրբ պատկերաւն, եւ առ ժամայն բժշկէին յախտիցն. եւ ամենայն ախտաժետք գային ՚ի համբաւն ՚ի ննջման տիրուհւոյն, եւ լինէին նոյնժամայն ողջացեալք, մինչ զի եւ համբաւն ընդ ամենայն Հրէաստան ընթանայր ծագման լուսոյն։    101 Արդ Բարթողոմէոս առաքեալ մի յերկոտասանիցն ոչ գոյր ընդ նոսա ՚ի զուգահաւաքման կուսին. վասն զի կամք էին տեառն այսմ այսպէս լինել, զի ընդ Թովմայի ՚ի հնդիկս էր երթեալ։ 102 Եւ եկեալ նա յետոյ առ առաքեալսն, բռնազբաւսեաց զնոսա, ցուցանել նմա զգանձն աստուածային զմարմին ամենաւրհնեալ կուսին։ 103 Եւ յաւժարեալ նոցա լնուլ զփափագ տենչանաց եղբաւրն իւրեանց, բացեալ այնուհետեւ զսուրբ գերեզմանն, գտին ունայն յաստուածակիր մարմնոյն։ 104 Եւ հիացեալք իմացան թէ ՚ի դադարել աւրհնութեան հրեշտակացն եւ սաղմոսերգութեանցն զկնի երից աւուրց ՚ի նոցանէ փոխադրեցաւ մարմինն սուրբ ՚ի ձեռն նոցին հրեղինաց դասուցն։ 105 Ապա յոյժ սգացեալ Բարթողոմէոս, զի ոչ ժամանեաց նա տեսութեան սուրբ կուսին, եւ ոչ լուաւ զաւրհնութիւնսն վերջին որպէս զայլ առաքեալսն եւ զաշակերտսն տեառն, եւ ոչ ՚ի գերեզմանին եգիտ, տրտմէր առաւել քան զառաւել։    106 Զայն տեսեալ երանելի առաքեալքն, մխիթարել զեղբայրն իւրեանց կամեցեալ, ետուն նմա զպատկեր Տիրամաւրն որ էր ՚ի փայտն նկարեալ, բնաւորեալ զաւրութեամբն։ 107 Զոր առեալ Բարթողոմէոսի զուարճացաւ, եւ իբրեւ ընդ կենդանւոյ համարեալ նայէր։ 108 Ապա մեծաւ փութով ել ՚ի Հայս հրամանաւ սուրբ հոգւոյն, զի զԹադէոս առաքեալ սպանեալ էր Սանատրուկ արքայ յԱրտազ գաւառի Հայոց։ 109 Իսկ նա եկեալ ՚ի Հայս, գնաց յերկիրն մոգուց եւ Պարսից. եւ հասեալ ՚ի քաղաք թագաւորութեան նոցա, որոյ այժմ ասի Խորասան բարբառովն իւրեանց. եւ արար ընդ մէջ կրակարանին սիւն լուսոյ հրեղէն, եւ կալեալ զսուրբ պատկերն ընդդէմ արեգականն, եւ արգել զճառագայթս նորա, եւ խաւարեցոյց զվեց ժամն. եւ ոչ հաւատացին, բայց միայն ութ ՚ի մոգուցն յայնցանէ, որք երկրպագեցին փրկչին, եւ եկեալ մկրտեցան յառաքելոյն, զորս հանեալ ընդ իւր ՚ի միջոյ ամբարշտացն, եւ բերեալ ՚ի Հայս։ 110 Եւ եկեալ յաշխարհն Անձաւացեաց ՚ի համբաւ քարի միոջ, քանզի դեւք բազումք բնակեալ էին անդ, եւ պատրէին զմարդիկ տեղւոյն, տուեալ յայնմ տեղւոջէ դեղս ախտականս առ ՚ի կատարել զպղծութիւնս ախտից, կռանաձայնս դարբնաց ահաւոր հրաշիւք արհաւիրս գործէին. յորս մարդիկ աշխարհին սովորեալք, անդ առ քրային դեգերէին, առեալ ՚ի չաստուածոցն ծրարս թարախածորս ՚ի պատիր ախտիցն, որպէս մեղաւորաց ի դժոխս։    20 Իսկ երկրորդի կողմն՝ գետ հրոյ յորդեալ ելանէ առաջի նորա։    21 Իսկ երրորդի կողմն՝ նովաւ լսին բարբառք դիմադարձացն եւ անհաւատիցն՝ եթէ երթայք, անիծեալք, ի հուրն յաւիտենից, որ պատրաստեալն է սատանայի եւ հրեշտակաց նորա։    22 Իսկ չորրորդի կողմն՝ ուրախալից ձայն. նովաւ կոչին արդարքն՝ եթէ եկայք աւրհնեալք Հաւր իմոյ, ժառանգեցէք զարքայութիւնն, պատրաստեալ ձեզ ի սկզբանէ աշխարհի։    23 Իսկ երկրորդի ստեղն՝ զորս գտեալ հրոյն, ճարակ եւ կերակուր իւր առնէ զամենեսեան։    24 Իսկ վերջին ստեղն՝ պատրաստէ սիրելեաց իւրոց բաժակ փրկութեան եւ անմահութեան յարքայութեանն երկնից։    25 Այս է պատճառք ձայնից երաժշտականաց։    26 Եւ եթէ ընդէր ութ կանոնիւք կարգեալ են Սաղմոսք Դաւթի, որով հանապազ կատարեմք զպաշտամունս եկեղեցւոյ, ինձ այսպէս թուի՝ թէ եւթն կանոնն՝ ըստ եւթն դարուն, իսկ ութն՝ ըստ ութ դարուն կարգեալ է. զոր եւ իմաստունն Սողովմոն ի զգաւնութեան խրատուն տարացոյց տայ յասելն՝ եթէ Տուր բաժինս եւթանց, այլ եւ ութից. եւ դարձեալ ի Սաղմոսն Դաւթի յասելն՝ եթէ Սաղմոս վասն ութերորդին։    27 Արդ, որպէս վերագոյն ասացաք, եթէ եւթն կանոն՝ յաղագս եւթն դարուն, իսկ ութերորդն՝ ըստ ութ դարուն խորհուրդն բերէ յինքեան, զոր եւ ի վերայ դարձի գերութեանն կարգեալ է ի Բաբելովնէ յԵրուսաղէմ. իսկ ըստ իմանալի տեսութեանն՝ յերկրէ յերկինս եւ յերկիր աւետեաց նախաշաւիղ ի վերելք։ 28 Վասն զի աւրինակ էր նոցայն մերոցս. Երուսաղէմն՝ դրախտին, Նաբուգաւդաւնոսոր՝ սատանայի, բաբելացիքն՝ դիւաց, երեք մանկունքն՝ մարդկային բնութեանս, հնոցն հրոյն՝ հանդերձեալ հրոյն։    29 Իսկ զերգս երից մանկանցն հանապազ առաւաւտին երգել յեկեղեցւոջ, զի որպէս նոքա գերեալք ի բաբելոնական բռնաւորէն ի հնոցն անկանէին, նոյնպէս եւ մեք ի մէջ հնոցի կենցաղոյս՝ գերեալք յաներեւոյթ բռնաւորէն, եւ ըմբռնեալք պէսպէս ախտիւք եւ լուցկիք եղեալ յաւիտենական հրոյն։ 30 Աստանաւր գոգցես իմն հրեշտակական ազդմամբ ի թառամեցուցանող քնոյն զարթուցեալք աւուր աւուր ընդ առաւաւտսն զերգս երից մանկանցն ի բերան առեալ՝ ասեմք եթէ Աւրհնեալ ես Տէր Աստուած հարցն մերոց, աւրհնեալ եւ փառաւորեալ անուն քո յաւիտեան։ 31 Եւ որպէս եւ նոքա ի փրկութիւն անձանց իւրեանց՝ առ իւրեանց փրկիչն մատուցեալ, եւ որպէս առ երկինս ի հնոցէն ի վեր գոչէին, յորդորելով զամենայն արարածս ի փառաբանութիւն Աստուծոյ. ըստ նմին աւրինակի եւ մեք՝ իբրեւ անդամք մարմնոյ նորա եւ ոսկերաց նորա՝ ի վերայ մեր եւ ի մեզ բերելով զգլուխն՝ վերագոչեմք աղաղակելով յընդհանուր տեսակս արարածոց՝ եթէ Աւրհնեցէք ամենայն գործք Տեառն զՏէր։ 32 Իբրեւ զի ազատեալք մեք ի մեղաց եւ ազատութեանն մերոյ հաղորդք լեալք ամենայն բնութիւնք արարածոց, հարկաւորին ամենայն իրաւք եւ նոքա ընդ մեզ աւրհնել եւ փառաւորել զԱստուած, զայն՝ որ ազատէն զամենեսին։    33 Բայց պարտ է նախկին ապաշխարութեամբ ի մեղացն արդարանալ, եւ ապա հրաման տալ արարածոցն ի փառաբանութիւն Աստուծոյ։ 34 Նախ ասել՝ եթէ Մեղաք, անաւրինեցաք, յանցեաք, եւ ապա համարձակել եւ ասել՝ եթէ Աւրհնեցէք ամենայն գործք Տեառն զՏէր. նախկին ասել՝ եթէ Ողորմեա ինձ, Աստուած, ըստ մեծի ողորմութեան քում, ըստ բազում գթութեան քում քաւեա զանաւրէնութիւնս իմ. առաւել լուա զիս, յանաւրէնութենէ իմմէ եւ ի մեղաց իմոց սուրբ արա զիս, եւ ապա համարձակել եւ ասել. Աւրհնեցէք զՏէր յերկնից, աւրհնեցէք զնա ի բարձանց. աւրհնեցէք զՏէր հրեշտակք նորա, աւրհնեցէք զնա ամենայն զաւրութիւնք նորա. եւ յետ այնորիկ բարձրանալ քան զզգալի արարածս յիմանալի բնութիւնսն, եւ ընդ անմարմնոցն դասուց գոչել՝ եթէ Փառք ի բարձունս Աստուծոյ եւ յերկիր խաղաղութիւն եւ ի մարդիկ հաճութիւն։ 35 Եւ այնպէս միաբանեալ ընդ վերնոցն՝ գեղեցիկ հակամիտութեամբ՝ նախ զսերովբէիցն զհնչումն երգեն զերեքկին սրբասացութիւնսն, եւ ապա զմանկունս եկեղեցւոյ յորդորելով՝ համաձայնել ընդ անմարմնական բնութիւնսն, յասելն՝ թէ Յարեւելից մինչեւ ի մուտս արեւու աւրհնեցէք զանուն Տեառն, որ գթացաւ առ աղքատ եւ տառապեալ բնութիւնսն, յարուցանելով յերկրէ եւ ի մեղաց եւ բնակեցուցանելով ի բարձունս յանմարմնական բնութիւնսն, փառաբանիչս ամենասուրբ Երրորդութեանն. նմա փառք յաւիտեանս յաւիտենից ամէն։ 36 Ողորմեա Տէր մեր։    ՄՈՎՍԷՍԻ ՔԵՐԹՈՂԱՀԱՒՐ՝ ÿԻ ՅԵՐԳՍ ԵՐԻՑ ՄԱՆԿԱՆՑՆ    1 Երեք մանկունքն աւրինակ էին մարդկային բնութեանս, եւ հնոց հրոյն՝ հանդերձեալ հրոյն։ 2 Իսկ զերգս երից մանկանցն հանապազ յառաւաւտուն երգել յեկեղեցւոջ. զի որպէս նոքա գերեալք ÿի բաբիլոնական բռնաւորէն ÿի հնոցն անկանէին, նոյնպէս եւ մեք ÿի հնոցի կենցաղոյս գերեալք յաներեւոյթ բռնաւորէն, եւ ըմբռնեալք պէս պէս ախտիւք, եւ լուցկիք եղեալ յաւիտենական հրոյն, աստանաւր գոգցես իմն հրեշտակական ազդմամբ ÿի թառամեցուցանաւղ քնոյն զարթուցեալք, աւուր աւուր ընդ առաւաւտն զերգս երից մանկանցն ÿի բերան առեալ ասեմք, Աւրհնեալ ես Տէր Աստուած հարցն մերոց. աւրհնեալ փառաւորեալ անուն քո յաւիտեան։    3 Եւ որպէս նոքա ÿի փրկութիւն անձանց իւրեանց առ իւրեանց փրկիչն մատուցեալ, եւ որպէս առ երկինս ÿի հնոցէն ÿի վեր գոչէին, յորդորելով զամենայն արարածս ÿի փառաբանութիւնն Աստուծոյ։ 4 Ըստ նմին աւրինակի եւ մեք իբրեւ անդամք ÿի մարմնոյ նորա եւ յոսկերաց նորա, ÿի վերայ մեր եւ ÿի մեզ բերելով զգլուխն, վերագոչեալ աղաղակեմք առ ընդհանուր տեսակս արարածոց թէ, Աւրհնեցէք ամենայն գործք Տեառն զՏէր։ 5 Իբր զի ազատեալք ÿի մեղացն՝ եւ ազատութեանն մերոյ հաղորդ լեալք ամենայն բնութիւնք արարածոց, հարկաւորին ամենայն իրաւք եւ նոքա ընդ մեզ աւրհնել եւ փառաւորել զԱստուած. զայն՝ որ ազատէն զամենեսեան։    6 Բայց պարտ է նախ ապաշխարութեամբ ÿի մեղացն արդարանալ, եւ ապա հրաման տալ արարածոց գալ ÿի փառաբանութիւնն Աստուծոյ։    7 Նախ ասել թէ, Մեղաք՝ անաւրինեցաք, յանցեաք։ 8 Եւ ապա համարձակիլ եւ ասել թէ, Աւրհնեցէք ամենայն գործք Տեառն զՏէր։ 9 Նախկին թէ, Ողորմեա ինձ Աստուած ըստ մեծի ողորմութեան քում. ըստ բազում գթութեան քում, քաւեա զանաւրէնութիւնս իմ։ 10 Առաւել լուա զիս յանաւրէնութէնէ իմմէ. եւ ÿի մեղաց իմոց սուրբ արա զիս։ 11 Եւ ապա համարձակիլ եւ ասել, Աւրհնեցէք զՏէր յերկնից, աւրհնեցէք զնա ÿի բարձանց։ 12 Աւրհնեցէք զնա ամենայն հրեշտակք նորա. աւրհնեցէք զնա ամենայն զաւրութիւնք նորա։
