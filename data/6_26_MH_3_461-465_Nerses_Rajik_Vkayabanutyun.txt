ՆԵՐՍԷՍ ՌԱԺԻԿՎԿԱՅԱԲԱՆՈՒԹԻՒՆ ՍՐԲՈՅՆ ՅԻԶՏԻԲՈՒԶՏԵԱՅ    1 Ընդ ժամանակս թագաւորութեանն Խոսրովու որդւոյ Կաւատայ արքային Պարսից` առաւել զայրացեալ լինէր խոստովանութիւնն որ առ Քրիստոս, եւ ոչ սակաւք նահատակեցան յաշխարհին Պարսից եւ ի դրան արքունի։ 2 Յորում ժամանակի այր ոմն Մախոշ մոգ` ի նորին աշխարհէ, ի գաւառէն, որ կոչի Բերշապուհ, ի գեղջէ, որոյ անուն Քունարաստան, որոյ ծնողքն պաշտաւնեայք էին կրակի, եւ ինքն ի մանկութենէ մոգութեամբ վարժեալ. եւ էր զաւրաւոր խրատու եւ գիտութեամբ հրապաշտիցն եւ խորհրդով իմաստուն։    3 Սա ի դիպուածոյ իմեքէ ի բանակս թագաւորին հասանէր, եւ պատահէր չարչարանաց սուրբ վկայի ուրումն` որում անուն էր Գրիգորիոս։ 4 Հրաշացեալ ընդ համարձակութիւն նահատակին, եւ զհաւատոց բարւոք նախանձն նախանձեալ` ասէ ÿի սրտի իւրում. Տէր Աստուած քրիստոնէից, նայեաց յիս եւ ցոյց ինձ ճանապարհ, որով գնացից, եւ բաց ինձ դուռն ողորմութեան ընդ նահատակի քում, զի արժանի եղէց արքայութեան քում, որ Տէրդ ես երկնի եւ երկրի։ 5 Իսկ ի նմին գիշերի տեսիլ սքանչելի երեւէր նմա, որպէս թէ լինէր ինքն ի սուրբ եկեղեցւոջ, եւ այր ոմն սպիտակազգեստ երեւեալ նմա ասէ. Երանելի ես դու, զի այսպիսի լուսաւոր եկեղեցւոյ եղեր արժանի. եւ հաւատասցես յայսմհետէ յԱստուած յարարիչն երկնի եւ երկրի եւ յՈրդին համազաւր եւ ի հոգին համագործ, եւ ուսցիս զայդ յաւանդութենէ սրբոյ Աւետարանին։ 6 Եւ զարթուցեալ իմացաւ զտեսիլն, եւ հաւատաց բոլորով սրտիւ եւ ասէ. Աստուածն քրիստոնէից է ճշմարիտ Աստուած։ 7 Եւ յայնմհետէ զղջմամբ ունէր զմոգութիւնն։    8 Գայր անտի յաշխարհն Սիւնեաց, պարապելով յունկնդրութիւն Գրոց Սրբոց. եւ լեալ անդ սակաւ ամիսս հասանէ ի Հայս. եւ եկեալ յԱյրարատ գաւառ` բնակէր յոստանին Հայոց ի Դուին, յարեալ ի մոգս ուրումն Հայոց համակարի, որում անուն էր Խոսրով Պերոզ. եւ արհամարհանաւք զմոգութիւնն անցուցանէր։ 9 Եւ լինէր ի ձմերային ժամանակի, ի մեհեկանին, ըստ Պարսկաց ամսոյ սկիզբն, եւ ըստ սովորութեանն իւրեանց զոհս աղտէղ մատուցանէին հրոյն յապարանս համակարին։ 10 Եւ երանելին Մախոշ կայր ընդ նոսա յառակի, պնդեալ զբերանն։ 11 Եւ բարձրացեալ բոցն ի ճարպոյն, հասանէր ի դարաւանդակ տանն եւ բորբոքէր։ 12 Իսկ մոգուցն ժողովեալ զզոհելն` զհուրն փութացան շիջուցանել, եւ ոչ կարացին. եւ արտաքս ընթացեալ պատմէին զիրսն։ 13 Եւ որ հազարապետն էր Հայոց, իբրեւ լուաւ զձայն ամբոխին, փութապէս հասանէր ÿի դուռն եկեղեցւոյն, ստիպէր զպաշտաւնեայսն առնուլ զսուրբ խաչն եւ բերել յապարանսն, զի թերեւս մերժեսցի հուրն, որ առ նեղութեանն ակամայ հաւատայր։ 14 Եւ սպասաւորացն առեալ զնշանս սուրբ խաչին` արագ հասանէին ի տեղին, եւ հասեալ յանդիման ապարանիցն սուրբ խաչն, եւ բոցն տարածեալ շիջանէր առ հասարակ։ 15 Այս մեծ նշան եղեւ պատուական սուրբ խաչիւն յաշխարհիս Հայոց` ÿի խնդութիւն բարեպաշտից եւ յամաւթ կռապաշտութեան։ 16 Եւ տեսեալ կնոջ համակարին` մեծ գոհութիւն առաքէ սպասաւորաց սքանչելագործ սուրբ խաչին։    17 Իսկ երանելին Մախոշ իմացեալ զայսպիսի սքանչելիս, զուռն որովք հրամայէր` ջարդեալ ի բաց ընկենոյր, նոյնպէս եւ զբերանակապն. եւ մատուցեալ առնոյր զբուրվառն ի ձեռաց սարկաւագին, եւ արկանէր խունկս առաջի սրբոյ խաչին, եւ յայտնէր զհաւատսն, զոր ունէր ի Քրիստոս. եւ յուղարկելով զսպասաւորս խաչին ասէր. Խնդրէք յԱստուծոյ վասն իմ, զի եւ ես քրիստոնեայ եմ։ 18 Եւ տեսեալ մոգուցն` ասեն ցնա. Ընդէր զայդ իշխեցեր առնել։ 19 Ասէ երանելին. Ես ի բազում ժամանակաց հաւատացի քրիստոնէից, եւ խղճիւ մոգէի, բայց արդ յայտնապէս քարոզեմ զհաւատս, որ ի Քրիստոս, եւ այլ ոչ եւս սպասաւորեցից հրոյ` զայդ ամենայն զարմանալիս տեսեալ։    20 Ի վաղիւ անդր երթեալ մոգուցն առաջի Հայոց իշխանին, որ կոչէր Նիխորական, եւ ամբաստանէին զերանելւոյն։ 21 Եւ ածեալ զնա առաջի` ասէ դատաւորն. Ա՜յ թշուառական, ընդէր զհայրենի մոգութիւնսն թողուս, եւ յիմար եւ ուխտակորոյս լինիս։ 22 Պատասխանի ետ երանելին եւ ասէ. Ճշմարտիւ ասեմ, ով դատաւոր, վասն իմ, զի քսան եւ հինգ ամ ունիմ, որով ընդ վայր աղանդաւորեցայ. իսկ այժմ ծանուցեալ զճշմարիտն Աստուած` ի հայրենի բազմաստուածութենէն հրաժարեցի։ 23 Վասն որոյ ոչ այսուհետեւ կենցաղոյս պատրանք յայսց հաւատոց անջատեսցեն զիս, քանզի հանդերձեալ եմ տանել հասելոց տանջանաց։ 24 Եւ զայրացեալ իշխանն` հրամայեաց հարկանել զերանելին երկոտասան կրկին, եւ կտրեալ զմաւրուսն եւ զհերս, եւ գրեալ զհարցումն` կապէր զոտս եւ զձեռս եւ տայր ի բանտ։ 25 Եւ գտանէր սուրբն իւր վշտակից զոմն ի բանտին, որում անուն էր Ներսէս, որ ի բազում ժամանակաց էր կապեալ ի բանտին վասն Աստուծոյ։ 26 Եւ ուսանէր ի նմանէ երանելին զսուրբ հաւատս եւ զսաղմոսս քսան, եւ այսպէս անդադար պաշտէր զԱստուած եւ աղաւթէր։    27 Յետ սակաւ ժամանակի փոխանորդէ զկնի Նիխորականին Նախապետն անուանեալ. եւ նստեալ յատենի` ածին զերանելին Մախոշ առաջի նորա։ 28 Ասէ Նախապետն. Զիարդ իշխեցեր զհայրենի աստուածն թողուլ եւ հաւատացեր անծանաւթին Քրիստոսի։ 29 Ասէ երանելին. Մի է Աստուած` հասարակաց արարիչ, որ ունի զմտաց ամենայն բարեգործաց, որոյ տեսեալ զմիտս իմ թուլացեալ ի բազմաստուածոցն երկրպագութենէ` առաջնորդեաց ինձ ի ճշմարիտ աստուածգիտութիւն։ 30 Եւ սրտմտեալ Նախապետին` հրամայեաց գան հարկանել զնա սաստիկ, եւ կապեալ նորին երկաթովք տարան ի բանտ։ 31 Եւ լեալ ի բանտին այլ եւս ամս երիս` հայցէր ի քահանայէն առնուլ զտէրունական կնիքն, զի անուանեսցէ զնա Յիզտիբուզիտ, որ թարգմանի Աստուածատուր, այսինքն է` յԱստուծոյ փրկեալ, զոր եւ Հոգին Սուրբ կենդանագրէր, որ եւ զաւրացեալ լինէր ի բանտին հոգւովն, եւ հանեալ տայր զհանդերձս իւր կարաւտելոց, եւ ինքն զգենոյր այծեայս եւ հողաթափս։ 32 Եւ այսպէս պահովք եւ աղաւթիւք հսկէր սաղմոսելով անդադար գեղեցիկ կրաւնաւորութեամբ։    33 Եւ ի լնուլ այլ եւս երից ամացն` յաջորդէր զՆախապետն որում Քնարիկն ասէին։ 34 Դպիր ոմն սորա երթեալ ի բանտն` հնարէր պատրել զերանելին. ասէ ցնա. Քեզ միայն Արեաց աշխարհն չեղեւ բաւական, զի մոլորեալ ես ի մոգութենէ, անձամբ զքեզ գերի վարելով առ հասարակ յաշխարհէ եւ ի դենէ մոլորեալ։ 35 Պատասխանի տուեալ սրբոյն ասէ. Ամենայն անծանաւթք ճարտարաց բնութեամբ սպասաւորին, եւ կենդանիքս ի ստացողէն ստրկացան. այսպէս եւ մարդիկ իւրեանց արարչին պարտին ծառայել միայն Աստուծոյ. վասն որոյ եւ ես յանտեսութեան իմում նանրութեան ծառայեցի. իսկ յորժամ գթացաւ յիս Աստուած, կալեալ զմտածութենէ իմմէ կապեաց ի սէր պատուիրանաց իւրոց` ներկեալ յուսով շնորհի իւրոյ, զոր ոչ կարասցեն կենցաղոյս փորձութիւնքս փոփոխել։ 36 Եւ լուեալ զայս ատենադպրին` յոտն յարեաւ ցասմամբ, եւ սկսաւ հարկանել զնա բրով անողորմ։ 37 Եւ այլ եւս բազում ազգի ազգի տանջանս հասուցանէր երանելւոյն, զի հաւանեսցի ունել զմոգութիւնն։    38 Յայնմ ժամանակի լուեալ թագաւորն զբազում բռնութիւն եւ զտառապանս զոր հասուցանէին աշխարհին Հայոց գործակալքն, առաքէր երիս իշխանս հաւատարիմս այց առնել երկրին, անուն միումն Նատոյ` որ կոչի Դրովանդական, եւ երկրորդին Պերոզ, որ էր Ռէոյ գաւառին մոգպետ, եւ երրորդին Խոյապ, որ էր սպասաւոր արքային. որք եկեալ ի Հայս բազում ուղղութիւնս արարին։ 39 Եւ յաւուր միում հրապարակաժողով արարեալ` նստէին յատենի յաղագս հրամանին արքունի։ 40 Ի նմին ժամու արք ամպարիշտք զգացուցին իշխանացն եւ ասեն. Են արք երեք ի բանտի, որոց զմեր պաշտաւնն թողեալ եւ զքրիստոնէութիւն են յանձն առեալ, Ներսէս ոմն Ռաժիկ, եւ Սահակ ոմն յԱտրպատականէ, եւ Յիզտիբուզիտ Պարսիկ։ 41 Եւ հրամայեցին ածել զնոսա։ 42 Ասէ մոգպետն ցՆերսէս. Ընդէր եղեր քրիստոնեայ։ 43 Ասէ երանելին. Մայր իմ քրիստոնեայ էր, եւ ի տղայութեանն էի հաւատացեալ ի Քրիստոս, չգիտեմ բնաւ զձեր դենդ։ 44 Հարցեալ եւ ընդ Սահակ. եւ նա ասէ. Ի տղայութեանն լեալ էի քրիստոնեայ։    45 Իսկ երանելին Յիզտիբուզիտ կայր յետոյ, եւ նուաղ ձայնիւ սաղմոսէր զվեցերորդ սաղմոսն. Տէր, մի սրտմտութեամբ քով յանդիմաներ զիս, եւ մի բարկութեամբ քով խրատեր զիս։ 46 Եւ զատուցեալ զնոսա` ասէ ցՅիզտիբուզիտ. Այ թշուառական, ընդէր զլուսաւոր դենն թողեր եւ նսեմաւոր քրիստոնէութեան հաւատացեր, եւ դարձեալ ստիպելով զոհեալ քո հրոյ։ 47 Ասէ սուրբն. Այո, զոհեալ, եւ զմոգութիւն առաւել գիտեմ քան զքեզ. եւ եթէ հրամայես ասել եւ յայտ առնել, ծանիցես որքան լաւ է քրիստոնէութիւն քան զամենայն պաշտամունս, զոր ես քննեալ եւ հաւաստեաւ ծանուցեալ, քրիստոնէութեան միայն է կարողութիւն փրկել յահագին ի մահուանէն։ 48 Եւ լուեալ մոգպետին` հրամայէր սպասաւորացն պատառեալ զպարեգաւտս նորա եւ հանել զկնգուղն. հարկանէին զգլուխ երանելւոյն եւ զմաւրուսն խլեալ արմատաքի։    49 Եւ դարձեալ զերեսեան երանելիսն հրամայեցին պահել զերիս աւուրս. եւ յետ այնորիկ ածեալ զնոսա` ոչ կարէին հաւանեցուցանել. ապա արգելին զնոսա ի տան միում անկերակուր։ 50 Եւ հրամայեաց մոգպետն արկանել ծուխ զգիշերն ամենայն։ 51 Վտանգեալ սուրբքն` աղաղակէին եւ ասէին. Աստուած, յաւգնել մեզ նայեաց, եւ Տէր, յընկերել մեզ փութա։ 52 Եւ այսպէս անդադար սաղմոսէին մինչեւ ցառաւաւտն. եւ լեալ ի բանտին աւուրս տասն։ 53 Եւ ի գիշերի միում յաւուր կիւրակէի, յորում հանդերձեալ էր Յիզտիբուզիտ վկայել, տեսութիւն երեւէր նմա. այր ոմն ունէր ի ձեռինն վարոց, եւ հասուցեալ ի գլուխ նորա` եւ հոսէր արիւնն ի միջէ աչաց նորա։ 54 Եւ երեւեալ երից լուսազգեաց արանց` ի բաց մերժեցին զայրն, որ եհարն, եւ բերեալ պսակ յընտիր մարգարտէ` դնէին ի գլուխ նորա։ 55 Եւ զարթուցեալ պատմեաց ընկերացն, եւ գիտացին եթէ հասեալ է ժամ նորա։    56 Եւ ի վաղիւն եկեալ հաւատարիմքն երթեալ ի բանտն ածել զսուրբսն. եւ եկեալ դահճապետն ի բանտն` տանել, եւ նա հրաժարեալ յեղբարցն գնայր. եւ յերթալն սաղմոսէր զչորրորդ սաղմոսն, սկսեալ ի բանտէն մինչեւ ի տեղի ատենին. Ի կարդալ իմում լուար ինձ ըստ արդարութեան, ի նեղութենէ ինձ անդորր արարեր. եւ կատարեալ զսաղմոսն` հասանէր ի տեղի ատենին։ 57 Եւ լուեալ քրիստոնէիցն` առ հասարակ դիմեցին անդր տեսանել զնահատակութիւն սրբոյն։ 58 Եւ սուրբն Յիզտիբուզիտ կայր առաջի չորից դատաւորացն, եւ կազմեալ երիս փայտս մահապարտաց. զոր բերեալ առաջի` ասէ մոգպետն ցսուրբն. Տեսանես զփայտսդ` յորում հանդերձեալ ես մեռանել. խնայեա ի քեզ, եւ մի ի նմին յամառութեան կար։ 59 Պատասխանի ետ սուրբն եւ ասէ. Առաւել եւս հաստատեալ եմ ի Քրիստոս, որ է յոյսն իմ. բայց դու զհրամայեալսն քեզ վասն իմ` կատարեա։ 60 Եւ հրամայեաց տանել ի փայտն. եւ սուրբն յաւժարութեամբ սրտիւ եւ զուարթ մտաւք երթայր ի փայտն. եւ հանեալ ի նմանէ ի բաց զերծուցին զհանդերձիկն, զոր ունէր։ 61 Ասէ երանելին. Յիրաւի է այս այսպէս, զի իմ Տեառնորդին ի կամաւոր խաչելութեանն Միածինն փրկիչ՝ մերկ ի խաչին բեւեռեցաւ, զի զնախահաւրն մերոյ զմերկութիւնն զգեցուսցէ, որ եւ զիս զաւրացուսցէ, զի պարտեցից զսատանայական խորամանկութիւնս ի մարմնի իմում։ 62 Եւ տարածեալ զինքն ի վերայ փայտին` սկսաւ ասել զքառասներորդ երրորդ սաղմոսն. Արի, Տէր, աւգնեա մեզ եւ փրկեա զմեզ վասն անուանդ քո. համարեցաք որպէս զոչխար ի զենումն։    63 Եւ ի կանգնել զփայտն` յղէ առ նա մոգպետն, զի թերեւս դարձցի։ 64 Եւ արհամարհեալ չտայր պատասխանի. կայր ի փայտին պնդեալ ի բազմամբոխ ատենին, եւ խոցէին նետիւք զկողս նորա։ 65 Դիպեցաւ ի նմին աւուր հանել ի փայտ երկուս մահապարտս, զորս եւ հանին իսկ ընդ նոսա. մինն էր ի ժողովոյն Խուժկաց, զոր ածեալ ի փայտն` համբուրէր զոտս սուրբ վկային, եւ առեալ հող արիւնախառն, որ ի կաթուածոց վկային, ընկենոյր ի ծոց իւր։ 66 Եւ դահճապետն կռփելով զգլուխ նորա, հանին ընդ աջմէ վկային ի փայտն, եւ ի նա ձգեալ զնետն եւ այնպէս աւանդեաց զոգին։ 67 Իսկ միւսն Հրեայ էր. իբրեւ ածին ի փայտն, բողոքեաց առ դատաւորսն. Մի կորուսանէք, զմոգութիւն յանձն առնում. եւ չկալեալ միտ դատաւորացն` հանին զնա ի փայտն յահեկէ սրբոյն։ 68 Եւ երանելին ի մէջ երկուց մահապարտացն ի բարձր տեղւոջ ողջակիզեալ յանդիման արեգականն եւ ամենայն ժողովրդոցն։ 69 Որ ունէր խորհուրդ զարմանալի. որ եւ յիշատակեալ ի նա տէրունւոյն բան` ըստ ճշմարիտ խոստմանն, թէ` Որ հաւատայ յիս` զգործն զոր ես գործեմ եւ նա գործեսցէ, եւ մեծամեծ եւս քան զնոյնս։ 70 Եւ առաքեալ ասէ. Զորս յառաջն ճանաչէր` յառաջագոյն հրաւիրեաց կերպարանակից լինել պատկերի Որդւոյ իւրոյ։    71 Եւ այսպէս քաջութեամբ վկայեաց սուրբն Յիզտիբուզիտ ի քսաներորդի երրորդի ամին Խոսրովու թագաւորի, որ աւր երկու էր քաղոց ամսոյ, յաւուր կիւրակէի յերրորդ ժամու. եւ հրամայեաց պահել զմարմին երանելւոյն ի վերայ փայտին։ 72 Իսկ հաւատացելոցն ըստ յաջողելոյն Աստուծոյ առեալ զմարմին սրբոյն, եւ պատուական հանդերձիւք պատեալ, իջուցին ի հանգիստ պայծառ պատուով, տաւն կատարելով մեծաւ ուրախութեամբ. յորում այսպէս լոյս փայլեալ զուարճացոյց զանձինս ուղղափառ հաւատացելոց։ 73 Վասն զի ամենափրկչին Յիսուսի Քրիստոսի Տեառն մերոյ, հանդերձ Հաւր ամենակալի եւ Հոգւոյն Սրբոյ, վայելէ փառք, պատիւ, այժմ եւ միշտ։    74 Եւ ես` Ներսէս (ասէ վկայաբանն) չարչարակից եւ ընկերակից սրբոյն, արարի զպատմութիւնս ի փառս Քրիստոսի Աստուծոյ մերոյ։
